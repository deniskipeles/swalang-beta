// cmd/vm/main.go
package main

import (
	"fmt"
	"io/ioutil"
	"os"
	"path/filepath"

	// VM specific execution path
	"github.com/deniskipeles/pylearn/internal/lexer"
	"github.com/deniskipeles/pylearn/internal/parser"
	"github.com/deniskipeles/pylearn/internal/compiler"
	"github.com/deniskipeles/pylearn/internal/vm"

	// Shared object model and stdlib init side effects
	"github.com/deniskipeles/pylearn/internal/object"
	_ "github.com/deniskipeles/pylearn/internal/stdlib/pyhttp"
	_ "github.com/deniskipeles/pylearn/internal/stdlib/pyos"
	_ "github.com/deniskipeles/pylearn/internal/stdlib/pyhttpserver"
	"github.com/deniskipeles/pylearn/internal/stdlib/pysys"

	// DO NOT import interpreter here
	// "github.com/deniskipeles/pylearn/internal/interpreter"
)

func main() {
	// --- Capture Command Line Arguments & Initialize Sys ---
	// (This part remains the same, using pysys directly)
	goArgs := os.Args
	pylearnArgv := make([]object.Object, len(goArgs))
	for i, arg := range goArgs { pylearnArgv[i] = &object.String{Value: arg} }
	argvList := &object.List{Elements: pylearnArgv}
	pysys.InitializeSysModule(argvList) // sys module initializes itself and registers

	// --- Argument Handling & Script Directory ---
	filename := ""
	var scriptDirSetter func(string) // Function variable for setting dir context

	if len(goArgs) >= 2 {
		filename = goArgs[1]
		// Need a way to set the script directory context that works for BOTH
		// interpreter and potentially future VM file loading/imports.
		// Let's assume a function exists in a shared package, maybe 'core'?
		// For now, we'll skip setting it here for the VM path, as the VM
		// doesn't use it yet for file imports. The interpreter main sets it.
		// If VM needed it: core.SetCurrentScriptDir(filename) // Hypothetical shared func
		scriptDirSetter = func(path string) {
			// TODO: Move SetCurrentScriptDir out of interpreter package if needed by VM file imports
			// For now, this path setter is unused in the VM main.
			absPath, err := filepath.Abs(path)
			if err != nil { /* handle warning */ return }
			info, err := os.Stat(absPath)
			if err == nil && info.IsDir() { /* use absPath */ } else { /* use filepath.Dir(absPath) */ }
			// interpreter.SetCurrentScriptDir(...) // Would need a shared version
		}
		if scriptDirSetter != nil {
			scriptDirSetter(filename) // Call if we had a shared setter
		}

	} else {
		 fmt.Fprintf(os.Stderr, "Usage: %s <filename.py> [args...]\n", filepath.Base(goArgs[0]))
		 os.Exit(1)
	}


	// --- File Reading ---
	sourceBytes, err := ioutil.ReadFile(filename)
	if err != nil { fmt.Fprintf(os.Stderr, "Error reading file '%s': %v\n", filename, err); os.Exit(1) }
	source := string(sourceBytes)

	// --- Lexing & Parsing (Common to both) ---
	l := lexer.New(source)
	p := parser.New(l)
	program := p.ParseProgram()
	if len(p.Errors()) != 0 { printParserErrors(os.Stderr, p.Errors()); os.Exit(1) }

	// --- COMPILER AND VM EXECUTION ---

	// 1. Compile
	comp := compiler.New()
	errComp := comp.Compile(program)
	if errComp != nil {
		fmt.Fprintf(os.Stderr, "Compilation failed:\n%s\n", errComp)
		os.Exit(1)
	}
	fmt.Println("--- Compilation Finished ---") // DEBUG

	 // 2. Get Bytecode and Print it for Debugging
	 bytecode := comp.Bytecode()
	 // +++ DEBUG: Print the generated bytecode +++
	 fmt.Println("--- Generated Bytecode ---")
	 fmt.Printf("Constants (%d):\n", len(bytecode.Constants))
	 for i, cnst := range bytecode.Constants {
		  fmt.Printf("  [%d]: %s\n", i, cnst.Inspect())
	 }
	 fmt.Println("Instructions:")
	 // Use the String() method from internal/code/code.go which should format it
	 fmt.Println(bytecode.Instructions.String())
	 fmt.Println("--------------------------")
	 // +++ END DEBUG +++
 
 
	 // 3. Set up VM
	 globals := make([]object.Object, vm.GlobalsSize) // Globals managed by VM
	 // *** ENSURE `bytecode` is passed here ***
	 machine := vm.NewWithGlobalsStore(bytecode, globals) // Correctly passed
 
	 // --- Populate VM's native modules ---
	 registeredModules := object.GetAllRegisteredModules()
	 machine.SetNativeModules(registeredModules)
	 // --- END POPULATE ---
 
	 // 4. Run VM
	 fmt.Println("--- Running VM ---")
	 errRun := machine.Run()
	 if errRun != nil { /* ... error handling ... */ os.Exit(1) }
 
	 // 5. Get Result (Optional)
	 result := machine.LastPoppedStackElem()
	 fmt.Println("--- VM Finished ---")
	 if result != nil && result != object.NULL {
		  fmt.Println("Final Stack Elem:", result.Inspect())
	 }
 
	 os.Exit(0)
}

// printParserErrors helper function
func printParserErrors(out *os.File, errors []string) {
    fmt.Fprintln(out, "Parser errors encountered:")
    for _, msg := range errors {
        fmt.Fprintf(out, "\t%s\n", msg)
    }
}
