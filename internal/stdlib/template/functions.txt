// pyTemplateNewFn: Creates a new, unparsed Template object
func pyTemplateNewFn(ctx object.ExecutionContext, args ...object.Object) object.Object {
	if len(args) < 1 || len(args) > 3 {
		return object.NewError(constants.TypeError, "template.new() takes 1-3 arguments (name, syntax_type, base_path), got %d", len(args))
	}
	nameObj, okName := args[0].(*object.String)
	if !okName { return object.NewError(constants.TypeError, "template.new() 'name' must be string") }
	name := nameObj.Value

	syntax := "go" // Default
	// var basePathArg object.Object = object.NULL

	// if len(args) >= 2 { basePathArg = args[1] } // Corrected: args[1] is syntax_type
	// if len(args) == 3 { basePathArg = args[2] } // Corrected: args[2] is base_path

	// Process syntax_type from args[1] if present
	if len(args) >= 2 && args[1] != object.NULL {
		syntaxObj, okSyntax := args[1].(*object.String)
		if !okSyntax { return object.NewError(constants.TypeError, "template.new() 'syntax_type' must be string or None") }
		sVal := strings.ToLower(syntaxObj.Value)
		if sVal == "pylearn_custom" || sVal == "pylearn" || sVal == "jinja_like" { syntax = "pylearn_custom"
		} else if sVal != "go" { return object.NewError("ValueError: template.new() unsupported 'syntax_type': %s", sVal) }
	}
	
	var pylearnLoader *TemplateLoader
	var basePathString string
	// Process base_path from args[2] if present
	if len(args) == 3 && args[2] != object.NULL { // args[2] is basePath
	    basePathValObj := args[2]
		bpStrObj, okPath := basePathValObj.(*object.String)
		if !okPath { return object.NewError(constants.TypeError, "template.new() 'base_path' must be string or None") }
		basePathString = bpStrObj.Value
		if basePathString != "" {
			pylearnLoader = NewTemplateLoader(basePathString)
		}
	}


	pyTmpl := &Template{ Name: name, syntaxType: syntax, loader: pylearnLoader }

	if syntax == "pylearn_custom" {
		cfg := config.NewConfig() // Corrected: use config.NewConfig()
		var fsLoader loaders.Loader
		if pylearnLoader != nil { // basePathString was used to create pylearnLoader
			var err error
			fsLoader, err = loaders.NewFileSystemLoader(pylearnLoader.basePath) // Use loader's basePath
			if err != nil { return object.NewError("TemplateLoaderError: Gonja FS loader for '%s': %v", pylearnLoader.basePath, err) }
		}
		pyTmpl.gonjaEnv = gonja.NewEnvironment(cfg, fsLoader)
		// Actual template parsing (fromString/fromFile) happens in Template.parse/load methods
	} else { // "go" syntax
		pyTmpl.goHTMLTmpl = gohtmltemplate.New(name)
		// FuncMap applied during parse/load for Go templates
	}
	return pyTmpl
}

// pyTemplateParseStringFn: Creates and parses a template from a string
func pyTemplateParseStringFn(ctx object.ExecutionContext, args ...object.Object) object.Object {
	if len(args) < 1 || len(args) > 2 { return object.NewError(constants.TypeError, "template.parse_string() takes 1 or 2 args")}
	templateStrObj, okStr := args[0].(*object.String)
	if !okStr { return object.NewError(constants.TypeError, "template.parse_string() template_string must be string")}

	var syntaxArg object.Object = object.NULL
	if len(args) == 2 { syntaxArg = args[1] }
	
	// Create a new Template object using template.new (which sets up the engine)
	// Name it anonymously, no base_path for string parsing.
	newTmplObj := pyTemplateNewFn(ctx, object.NewString(""), syntaxArg, object.NULL)
	if object.IsError(newTmplObj) { return newTmplObj }
	
	tmplToParse := newTmplObj.(*Template)

	// Call the parse method on this newly created Template object
	return pyTemplateParseMethod(ctx, tmplToParse, templateStrObj)
}

// pyTemplateLoadFn: Creates and loads/parses a template from a file
func pyTemplateLoadFn(ctx object.ExecutionContext, args ...object.Object) object.Object {
	if len(args) < 1 || len(args) > 3 { return object.NewError(constants.TypeError, "template.load() takes 1-3 args")}
	filenameObj, okFile := args[0].(*object.String)
	if !okFile { return object.NewError(constants.TypeError, "filename for load must be string")}
	
	var syntaxArg, basePathArg object.Object = object.NULL, object.NULL
	if len(args) >= 2 { syntaxArg = args[1] }
	if len(args) == 3 { basePathArg = args[2] }

	// Create a Template object using template.new, passing the filename as its name
	// and any provided syntax_type and base_path.
	// template.new will correctly set up the loader if base_path is given.
	newTmplObj := pyTemplateNewFn(ctx, filenameObj, syntaxArg, basePathArg)
	if object.IsError(newTmplObj) { return newTmplObj }

	tmplToLoadInto := newTmplObj.(*Template)
	
	// Call the load method on this newly created Template object
	// pyTemplateLoadMethod expects (self, filename_obj_for_loading)
	return pyTemplateLoadMethod(ctx, tmplToLoadInto, filenameObj)
}


// pyTemplateRenderFn and pyTemplateRenderFileFn (no changes to their internal logic)
func pyTemplateRenderFn(ctx object.ExecutionContext, args ...object.Object) object.Object {
	if len(args) < 2 || len(args) > 3 { return object.NewError(constants.TypeError, "template.render()...") }
	templateStrObj := args[0]; dataObj := args[1]; var syntaxTypeObj object.Object = object.NULL
	if len(args) == 3 { syntaxTypeObj = args[2] }
	tmplObj := pyTemplateParseStringFn(ctx, templateStrObj, syntaxTypeObj) // This now returns a parsed *Template
	if object.IsError(tmplObj) { return tmplObj }
	return pyTemplateExecuteMethod(ctx, tmplObj, dataObj) // Pass dataObj
}
func pyTemplateRenderFileFn(ctx object.ExecutionContext, args ...object.Object) object.Object {
	if len(args) < 2 || len(args) > 4 { return object.NewError(constants.TypeError, "template.render_file()...")}
	filenameObj := args[0]; dataObj := args[1]; 
	var syntaxTypeObj, basePathObj object.Object = object.NULL, object.NULL
	if len(args) >= 3 { syntaxTypeObj = args[2] }
	if len(args) == 4 { basePathObj = args[3] }
	tmplObj := pyTemplateLoadFn(ctx, filenameObj, syntaxTypeObj, basePathObj) // This now returns a loaded *Template
	if object.IsError(tmplObj) { return tmplObj }
	return pyTemplateExecuteMethod(ctx, tmplObj, dataObj) // Pass dataObj
}

var (
	TemplateNew         = &object.Builtin{Name: "template.new", Fn: pyTemplateNewFn}
	TemplateParseString = &object.Builtin{Name: "template.parse_string", Fn: pyTemplateParseStringFn}
	TemplateLoad        = &object.Builtin{Name: "template.load", Fn: pyTemplateLoadFn}
	TemplateRender      = &object.Builtin{Name: "template.render", Fn: pyTemplateRenderFn}
	TemplateRenderFile  = &object.Builtin{Name: "template.render_file", Fn: pyTemplateRenderFileFn}
)
