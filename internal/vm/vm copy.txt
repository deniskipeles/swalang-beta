// ... (Keep consts, VM Types, Frame, ExceptionHandler, VM struct) ...
const StackSize = 2048
const GlobalsSize = 65536
const MaxFrames = 1024

// --- VM-Specific Types (Closure, Class, Instance, BoundMethod) ---
// VM Closure holds *core.CompiledFunction
type Closure struct {
	Fn   *core.CompiledFunction // Points to the shared core.CompiledFunction
	Free []object.Object
}

func (c *Closure) Type() object.ObjectType { return core.VM_CLOSURE_OBJ } // Use core type if available
func (c *Closure) Inspect() string         { return fmt.Sprintf("<vm.Closure Fn=%s>", c.Fn.Inspect()) }

var _ object.Object = (*Closure)(nil)

// VM Class holds vm.Closure methods
type Class struct {
	Name    string
	Methods map[string]*Closure // Stores vm.Closure
	// TODO: Add Superclass *Class later
}

func (c *Class) Type() object.ObjectType { return core.VM_CLASS_OBJ } // Use core type if available
func (c *Class) Inspect() string         { return fmt.Sprintf("<vm.Class '%s'>", c.Name) }

var _ object.Object = (*Class)(nil)

// VM Instance - separate from object.Instance
type Instance struct {
	Class *Class // Holds *vm.Class
	Env   *object.Environment
}

func (i *Instance) Type() object.ObjectType { return object.INSTANCE_OBJ } // Use base type name
func (i *Instance) Inspect() string         { return fmt.Sprintf("<%s object at %p>", i.Class.Name, i) }

var _ object.Object = (*Instance)(nil)

// VM BoundMethod holds vm.Closure
type BoundMethod struct {
	Instance object.Object // Can hold vm.Instance (usually)
	Method   *Closure      // Stores vm.Closure
}

func (bm *BoundMethod) Type() object.ObjectType { return core.VM_BOUND_METHOD_OBJ } // Use core type if available
func (bm *BoundMethod) Inspect() string {
	iname := "?"
	if bm.Instance != nil {
		iname = bm.Instance.Inspect()
	}
	mname := "?"
	if bm.Method != nil && bm.Method.Fn != nil { // Add nil check for Fn
		mname = bm.Method.Fn.Name
		if mname == "" {
			mname = "<method>"
		}
	}
	cname := "?"
	if inst, ok := bm.Instance.(*Instance); ok && inst.Class != nil {
		cname = inst.Class.Name
	}
	return fmt.Sprintf("<vm.BoundMethod %s.%s of %s>", cname, mname, iname)
}

var _ object.Object = (*BoundMethod)(nil)

// --- VM Frame holds vm.Closure ---
type Frame struct {
	cl          *Closure // vm.Closure being executed in this frame
	ip          int      // Instruction Pointer *within this frame's code* (Points to the *next* instruction to execute)
	basePointer int      // Stack index *before* the first argument/local slot for this frame
}

func NewFrame(cl *Closure, basePointer int) *Frame {
	if cl == nil {
		panic("VM Error: Attempted to create frame with nil closure")
	}
	return &Frame{cl: cl, ip: -1, basePointer: basePointer} // Start IP at -1, first increment makes it 0
}
func (f *Frame) Instructions() code.Instructions {
	if f.cl == nil || f.cl.Fn == nil {
		panic(fmt.Sprintf("VM Error: Attempting to get instructions from nil closure or function in frame (bp=%d)", f.basePointer))
	}
	return f.cl.Fn.Instructions
}

// --- NEW: Exception Handler Info ---
type ExceptionHandler struct {
	ExceptTargetIP    int // Instruction Pointer to jump to on exception
	FinallyTargetIP int // Instruction Pointer for finally block (0 if none)
	StackPointerLevel int // SP before entering the try block
	FrameIndex        int // Frame index when handler was pushed
}



// --- VM Struct Definition ---
type VM struct {
	constants     []object.Object
	stack         []object.Object
	sp            int // Points to the *next free slot*
	globals       []object.Object
	frames        []*Frame
	frameIndex    int // Index of the *current* frame
	builtins      []*object.Builtin
	handlerStack  []*ExceptionHandler
	nativeModules map[string]*object.Module
	callbackChan  chan CallbackRequest
}


// --- Constructors and Setup ---
func New(bytecode *compiler.Bytecode) *VM {
	mainFn := &core.CompiledFunction{Instructions: bytecode.Instructions, Name: "<main>"}
	mainClosure := &Closure{Fn: mainFn}
	mainFrame := NewFrame(mainClosure, 0)

	// Setup builtins deterministically
	builtinNames := make([]string, 0, len(builtins.Builtins))
	for name := range builtins.Builtins { builtinNames = append(builtinNames, name) }
	sort.Strings(builtinNames)
	builtinObjs := make([]*object.Builtin, len(builtinNames))
	for i, name := range builtinNames { builtinObjs[i] = builtins.Builtins[name] }

	frames := make([]*Frame, MaxFrames); frames[0] = mainFrame

	return &VM{
		constants:     bytecode.Constants,
		stack:         make([]object.Object, StackSize),
		sp:            0,
		globals:       make([]object.Object, GlobalsSize),
		frames:        frames,
		frameIndex:    0,
		builtins:      builtinObjs,
		handlerStack:  make([]*ExceptionHandler, 0, 16),
		nativeModules: make(map[string]*object.Module),
		callbackChan:  make(chan CallbackRequest, 16),
	}
}

func (vm *VM) SetNativeModules(modules map[string]*object.Module) { vm.nativeModules = modules }
func NewWithGlobalsStore(bytecode *compiler.Bytecode, s []object.Object) *VM {
	vm := New(bytecode); vm.globals = s; return vm
}


// Add a helper for concise object inspection in logs
func objInspect(obj object.Object) string { // Keep this helper
	if obj == nil { return "<nil>" }
	insp := obj.Inspect()
	if len(insp) > 50 { insp = insp[:47] + "..." }
	return fmt.Sprintf("%s (%s)", insp, obj.Type())
}
func (vm *VM) currentFrame() *Frame { // Keep this helper
	if vm.frameIndex < 0 || vm.frameIndex >= MaxFrames { panic(fmt.Sprintf("VM Error: accessing frame with invalid index (%d)", vm.frameIndex)) }
	frame := vm.frames[vm.frameIndex]
	if frame == nil { panic(fmt.Sprintf("VM Error: current frame slot (index %d) is nil", vm.frameIndex)) }
	return frame
}

func (vm *VM) pushFrame(f *Frame) error { // Keep this helper
	if f == nil { panic("VM Error: Attempted to push a nil frame") }
	nextFrameIndex := vm.frameIndex + 1
	if nextFrameIndex >= MaxFrames { return fmt.Errorf("stack overflow: max frames exceeded (%d)", MaxFrames) }
	fmt.Printf(">>> DEBUG VM: pushFrame: Pushing Frame %d (%s), BP=%d. Old FrameIdx=%d\n", nextFrameIndex, f.cl.Fn.Name, f.basePointer, vm.frameIndex)
	vm.frameIndex = nextFrameIndex; vm.frames[vm.frameIndex] = f
	return nil
}

func (vm *VM) popFrame() *Frame { // Keep this helper
	if vm.frameIndex < 0 { panic("VM Error: popping frame below base") }
	poppedFrame := vm.frames[vm.frameIndex]
	if poppedFrame == nil { panic(fmt.Sprintf("VM Error: Popped a nil frame from index %d", vm.frameIndex)) }
	fmt.Printf(">>> DEBUG VM: popFrame: Popping Frame %d (%s), BP=%d. New FrameIdx=%d\n", vm.frameIndex, poppedFrame.cl.Fn.Name, poppedFrame.basePointer, vm.frameIndex-1)
	vm.frames[vm.frameIndex] = nil // Clear the slot
	vm.frameIndex--; return poppedFrame
}
func (vm *VM) push(o object.Object) error { // Keep this helper
	if vm.sp >= StackSize { return fmt.Errorf("stack overflow: push failed (sp=%d, size=%d)", vm.sp, StackSize) }
	if o == nil { o = object.NULL }
	vm.stack[vm.sp] = o; vm.sp++; return nil
}
func (vm *VM) pop() (object.Object, error) { // Keep this helper
	if vm.sp <= 0 { return nil, fmt.Errorf("stack underflow: pop failed (sp=%d)", vm.sp) }
	vm.sp--; o := vm.stack[vm.sp]; vm.stack[vm.sp] = nil
	if o == nil { o = object.NULL }; return o, nil
}
func (vm *VM) peek(offset int) (object.Object, error) { // Keep this helper
	peekIndex := vm.sp - 1 - offset
	if peekIndex < 0 { return nil, fmt.Errorf("stack peek failed: index %d < 0 (sp=%d, offset=%d)", peekIndex, vm.sp, offset) }
	if peekIndex >= StackSize { return nil, fmt.Errorf("stack peek failed: index %d >= StackSize %d (sp=%d, offset=%d)", peekIndex, StackSize, vm.sp, offset) }
	obj := vm.stack[peekIndex]; if obj == nil { obj = object.NULL }; return obj, nil
}
func (vm *VM) StackTop() (object.Object, error) { // Keep this helper
	if vm.sp == 0 { return object.NULL, nil }; return vm.peek(0)
}
func (vm *VM) LastPoppedStackElem() object.Object { // Keep this helper
	top, _ := vm.StackTop(); return top
}

func (vm *VM) isInstance(obj object.Object, classInfo object.Object) (bool, error) { // Keep this helper
	targetClass, ok := classInfo.(*Class)
	if !ok { return false, object.NewError("TypeError: isinstance() arg 2 must be a vm.Class, not %s", classInfo.Type()) }
	if errObj, isErr := obj.(*object.Error); isErr {
		errName := ""; if idx := strings.Index(errObj.Message, ":"); idx != -1 { errName = errObj.Message[:idx] } else { errName = errObj.Message }
		return errName == targetClass.Name, nil // TODO: Hierarchy
	}
	if _, isStopIter := obj.(*object.StopIterationError); isStopIter { return targetClass.Name == "StopIteration", nil } // TODO: Proper type check
	instance, ok := obj.(*Instance)
	if !ok { return false, nil } // Not a user instance
	if instance.Class == nil { return false, fmt.Errorf("internal error: vm.Instance has nil Class") }
	currentCls := instance.Class
	for currentCls != nil { if currentCls == targetClass { return true, nil }; break } // TODO: Superclass traversal
	return false, nil
}

// --- executeCallback (More Robust SP handling on error) ---
func (vm *VM) executeCallback(req CallbackRequest) object.Object {
    fmt.Printf(">>> DEBUG VM: executeCallback START for %s\n", objInspect(req.Callable)) // DEBUG
    initialSP := vm.sp // Record SP *before* any modification
    initialFrameIndex := vm.frameIndex // Record frame index
    callable := req.Callable
    args := req.Args

    // Defer SP check/restore
    defer func() {
        if vm.sp != initialSP {
            // This might happen if an error occurred and wasn't fully cleaned up
             fmt.Printf(">>> DEBUG VM: executeCallback DEFER detected SP Mismatch: Expected %d, got %d. Forcing restore.\n", initialSP, vm.sp)
             // Maybe force SP reset, but this could hide deeper issues.
             // For now, just log the warning.
             // vm.sp = initialSP
        }
         fmt.Printf(">>> DEBUG VM: executeCallback DEFER finished. Final SP: %d\n", vm.sp)
    }()


    // --- Call Setup ---
    err := vm.push(callable)
    if err != nil { return object.NewError("VM Callback Error pushing callable: %v", err) } // SP not modified yet if push fails
    // spAfterCallable := vm.sp
    for i, arg := range args {
        err = vm.push(arg)
        if err != nil {
            // Restore SP to *before* pushing callable if args push fails
            vm.sp = initialSP
            return object.NewError("VM Callback Error pushing arg %d: %v", i, err)
        }
    }

    var callErr error; numArgs := len(args)
    switch callee := callable.(type) {
		case *Closure: callErr = vm.callClosure(callee, numArgs)
		case *object.Builtin: callErr = vm.callBuiltin(callee, numArgs)
		case *Class: callErr = vm.callClass(callee, numArgs)
		case *BoundMethod: callErr = vm.callBoundMethod(callee, numArgs)
		default: callErr = object.NewError("TypeError: '%s' object is not callable (in VM callback)", callee.Type())
	}
    if callErr != nil {
        vm.sp = initialSP // Restore SP fully if setup failed
        if pyErr, ok := callErr.(object.Object); ok && object.IsError(pyErr) { return pyErr }
        return object.NewError("VM Callback Setup Error: %v", callErr)
    }

    // --- Check VM state *before* running ---
    if vm.frameIndex < 0 {
        vm.sp = initialSP // Restore SP
        // ___fmt.Println(">>> DEBUG VM:   executeCallback detected terminated VM before running frame.")
        return object.NewError("VM terminated before callback could execute")
    }

    // --- Execute ---
    callbackTargetFrameIndex := initialFrameIndex + 1
    fmt.Printf(">>> DEBUG VM: executeCallback running from frame %d\n", callbackTargetFrameIndex)
    runErr := vm.runSingleCallbackFrame(callbackTargetFrameIndex) // Executes until target frame is popped
    fmt.Printf(">>> DEBUG VM: executeCallback frame execution finished, runErr=%v\n", runErr)

    // --- Process Result ---
    var finalResult object.Object
    if runErr != nil {
        // Error during execution, SP might be inconsistent, restore it
        fmt.Printf(">>> DEBUG VM: executeCallback restoring SP due to runErr. Before: %d, Target: %d\n", vm.sp, initialSP)
        vm.sp = initialSP
        if pyErr, ok := runErr.(object.Object); ok && object.IsError(pyErr) { finalResult = pyErr } else { finalResult = object.NewError("VM Callback Execution Error: %v", runErr) }
    } else {
        // Execution succeeded, result should be on stack. Pop it.
        poppedResult, popErr := vm.pop()
        if popErr != nil {
             // Error popping result, restore SP
             fmt.Printf(">>> DEBUG VM: executeCallback restoring SP due to popErr. Before: %d, Target: %d\n", vm.sp, initialSP)
             vm.sp = initialSP
             finalResult = object.NewError("VM Callback Error popping result: %v", popErr)
        } else {
            finalResult = poppedResult
            // SP check is now deferred
        }
    }
    fmt.Printf(">>> DEBUG VM: executeCallback END. Result: %s, Final SP: %d, Initial SP: %d\n", objInspect(finalResult), vm.sp, initialSP)
    // SP check happens in defer now

    return finalResult
}






// --- runSingleCallbackFrame ---
// Executes instructions starting from the current IP of the *active* frame,
// continuing until the frame at `targetFrameIndex` is popped or an unhandled error occurs.
// Used by synchronous callbacks (e.g., builtins calling back into VM).
func (vm *VM) runSingleCallbackFrame(targetFrameIndex int) (runErr error) {
	if vm.frameIndex < targetFrameIndex || vm.frameIndex < 0 { // Add vm.frameIndex < 0 check
		fmt.Printf(">>> DEBUG VM: runSingleCallbackFrame detected invalid frame state (current: %d, target: %d)\n", vm.frameIndex, targetFrameIndex)
		return object.NewError("VM execution context invalid or already terminated")
	}
	fmt.Printf(">>> DEBUG VM: runSingleCallbackFrame START for target frame %d (current %d)\n", targetFrameIndex, vm.frameIndex)
	defer func() {
		fmt.Printf(">>> DEBUG VM: runSingleCallbackFrame END for target frame %d (current %d), runErr=%v\n", targetFrameIndex, vm.frameIndex, runErr)
		if r := recover(); r != nil { runErr = object.NewError("Panic recovered during callback execution: %v", r); 
		fmt.Printf(">>> DEBUG VM: PANIC RECOVERED in runSingleCallbackFrame: %v\n", r) 
	}
	}()
	if vm.frameIndex < targetFrameIndex { return fmt.Errorf("internal error: callback target frame %d already popped (current %d)", targetFrameIndex, vm.frameIndex) }

runSingleFrameLoop:
	for vm.frameIndex >= targetFrameIndex { // Loop while target or inner frames are active
		currentFrame := vm.currentFrame()
		instructions := currentFrame.Instructions()
		ipToExecute := currentFrame.ip + 1
		if ipToExecute < 0 || ipToExecute >= len(instructions) {
			fmt.Printf(">>> DEBUG VM: runSingleCallbackFrame IP %d out of bounds [0, %d) for Frame %d (%s).\n", ipToExecute, len(instructions), vm.frameIndex, currentFrame.cl.Fn.Name)
			if vm.frameIndex == targetFrameIndex {
				// ___fmt.Println(">>> DEBUG VM:   runSingleCallbackFrame reached end of target frame instructions.")
				// This frame finished without returning, treat as implicit return NULL
				poppedFrame := vm.popFrame() // Pop the target frame
                basePointer := poppedFrame.basePointer
                if basePointer < 0 { runErr = fmt.Errorf("internal error: invalid basepointer %d for callback implicit return", basePointer); break runSingleFrameLoop }
                if basePointer >= StackSize { runErr = fmt.Errorf("internal error: basepointer %d out of bounds for callback implicit return value", basePointer); break runSingleFrameLoop }
				vm.stack[basePointer] = object.NULL // Place NULL at caller's expected result slot
                vm.sp = basePointer + 1              // Adjust SP
				runErr = nil // No actual error
			} else {
				runErr = fmt.Errorf("internal vm error: IP went out of bounds in inner frame %d during callback without return", vm.frameIndex)
			}
			break runSingleFrameLoop
		}
		currentFrame.ip = ipToExecute
		op := code.Opcode(instructions[ipToExecute])
		opDef, err := code.Lookup(byte(op)); if err != nil { runErr = fmt.Errorf("internal vm error: unknown opcode %d ...", op); break runSingleFrameLoop }

		frameIdxBefore := vm.frameIndex
		ipBeforeExecute := ipToExecute
		opDefBeforeExecute := opDef

		ipWasSetByJump, execErr := vm.executeInstruction(currentFrame)

        // --- Handle Return Signals FIRST (within callback context) ---
        isReturnSignal := false
        if execErr != nil {
            if errors.Is(execErr, ErrReturnSignal) || errors.Is(execErr, ErrVoidReturnSignal) {
                 isReturnSignal = true
                 fmt.Printf(">>> DEBUG VM: runSingleCallbackFrame Return signal received from frame %d\n", frameIdxBefore)
                 // Pop the frame THAT signaled return
                 // The return value is already placed correctly by executeInstruction
                 vm.popFrame()
                 // If the popped frame WAS the target frame, the callback is done.
                 if frameIdxBefore == targetFrameIndex {
                     // ___fmt.Println(">>> DEBUG VM:   runSingleCallbackFrame returned from target frame.")
                     runErr = nil // Callback finished successfully
                     break runSingleFrameLoop
                 }
                 // If it was an inner frame, adjust caller's IP within the callback context
                 if vm.frameIndex >= targetFrameIndex { // Ensure caller is still within callback scope
                    callerFrame := vm.currentFrame()
                    callOpDef := opDefBeforeExecute
                    callInstructionWidth := 1
                    if callOpDef != nil && callOpDef.Name == "OpCall" {
                        for _, w := range callOpDef.OperandWidths { callInstructionWidth += w }
                        callerFrame.ip = ipBeforeExecute + callInstructionWidth - 1
                        fmt.Printf(">>> DEBUG VM: runSingleCallbackFrame Post-Return IP Adjust: Caller IP set to %d\n", callerFrame.ip)
                    } else {
                         fmt.Printf(">>> DEBUG VM: runSingleCallbackFrame WARNING - Return signal but instruction at %d wasn't OpCall?\n", ipBeforeExecute)
                    }
                 }
                 execErr = nil // Clear the signal error
                 continue runSingleFrameLoop // Skip default IP advance
            }
        }

		// --- Handle REAL Execution Errors (within callback context) ---
		if execErr != nil {
			fmt.Printf(">>> DEBUG VM: runSingleCallbackFrame Instruction %s at IP=%d resulted in error: %v. Handling...\n", opDefBeforeExecute.Name, ipBeforeExecute, execErr)
			var pyErrInstance error; if errObj, ok := execErr.(error); ok { _, isObjErr := execErr.(*object.Error); _, isStopIterErr := execErr.(*object.StopIterationError); if isObjErr || isStopIterErr { pyErrInstance = errObj } else { pyErrInstance = object.NewError("RuntimeError: %v", execErr) } } else { pyErrInstance = object.NewError("InternalError: Unknown non-error type returned: %T value: %v", execErr, execErr) }
			handlerFound := false
			// Only handle exceptions relevant to frames >= targetFrameIndex
			for len(vm.handlerStack) > 0 && vm.handlerStack[len(vm.handlerStack)-1].FrameIndex >= targetFrameIndex {
				handler := vm.handlerStack[len(vm.handlerStack)-1]; vm.handlerStack = vm.handlerStack[:len(vm.handlerStack)-1]; handlerFound = true; 
				fmt.Printf(">>> DEBUG VM: runSingleCallbackFrame found handler targeting frame %d, exceptIP=%d\n", handler.FrameIndex, handler.ExceptTargetIP)
				for vm.frameIndex > handler.FrameIndex && vm.frameIndex >= targetFrameIndex { if vm.frameIndex < targetFrameIndex { runErr = object.NewError("Internal VM Error: Frame index < target during callback unwind."); break runSingleFrameLoop }; 
				fmt.Printf(">>> DEBUG VM: runSingleCallbackFrame unwinding frame %d for exception\n", vm.frameIndex); vm.popFrame() 
			}
				if vm.frameIndex != handler.FrameIndex { runErr = object.NewError("Internal VM Error: Frame index mismatch after callback unwind (expected %d, got %d)", handler.FrameIndex, vm.frameIndex); break runSingleFrameLoop }
				currentFrame = vm.currentFrame()
				if handler.StackPointerLevel < 0 || handler.StackPointerLevel > vm.sp { runErr = object.NewError("Internal VM Error: Invalid SP level (%d vs current %d) in callback handler", handler.StackPointerLevel, vm.sp); break runSingleFrameLoop }
				fmt.Printf(">>> DEBUG VM: runSingleCallbackFrame unwinding stack from SP %d to %d\n", vm.sp, handler.StackPointerLevel); for i := handler.StackPointerLevel; i < vm.sp; i++ { vm.stack[i] = nil }; vm.sp = handler.StackPointerLevel
				var pyErrorObject object.Object; if objErr, ok := pyErrInstance.(object.Object); ok { pyErrorObject = objErr } else { pyErrorObject = object.NewError("Internal VM Error: Cannot push non-object error: %v", pyErrInstance) }
				errPush := vm.push(pyErrorObject); if errPush != nil { runErr = object.NewError("Internal VM Error: Stack overflow pushing exception object during callback: %v", errPush); break runSingleFrameLoop }
				fmt.Printf(">>> DEBUG VM: runSingleCallbackFrame pushed exception %s, Jumping to handler IP %d\n", objInspect(pyErrorObject), handler.ExceptTargetIP)
				targetIP := handler.ExceptTargetIP; if targetIP < 0 || targetIP >= len(currentFrame.Instructions()) { runErr = object.NewError("Internal VM Error: Invalid ExceptTargetIP %d in callback handler", targetIP); break runSingleFrameLoop }
				currentFrame.ip = targetIP - 1
				continue runSingleFrameLoop // Continue execution in the handler
			}
			if !handlerFound { 
				// ___fmt.Println(">>> DEBUG VM:   runSingleCallbackFrame no handler found, propagating error"); 
				runErr = pyErrInstance; break runSingleFrameLoop 
			}
		} // End Error Handling

		// --- Advance IP for NEXT cycle (if not handled by jump/return) ---
		if !ipWasSetByJump && !isReturnSignal {
			operandWidth := 0
			if opDefBeforeExecute != nil { for _, w := range opDefBeforeExecute.OperandWidths { operandWidth += w } }
			currentFrame.ip += operandWidth
            fmt.Printf(">>> DEBUG VM: runSingleCallbackFrame Normal Advance: IP set to %d\n", currentFrame.ip) // Optional noise reduction
		}

	} // End callback execution loop

	return runErr
}



// executeInstruction executes the instruction at currentFrame.ip.
// It modifies currentFrame.ip *only* for jumps.
// It returns ipWasSet=true if it handled control flow (jump, call, return),
// otherwise it returns ipWasSet=false.
// It also returns any execution error encountered.
// --- executeInstruction (using return signals) ---
// Used internally by executeInstruction to signal returns to the Run loop.
var ErrReturnSignal = errors.New("signal: return value")
var ErrVoidReturnSignal = errors.New("signal: void return")
func (vm *VM) executeInstruction(currentFrame *Frame) (ipWasSetByJump bool, execErr error) {
	ip := currentFrame.ip
	instructions := currentFrame.Instructions()
	op := code.Opcode(instructions[ip])
	opDef, err := code.Lookup(byte(op))
	if err != nil { return false, fmt.Errorf("internal VM error: unknown opcode %d at ip %d", op, ip) }
	opName := opDef.Name

	ipWasSetByJump = false // Default: Run loop advances IP

	var readOperands []int
	numOperandBytes := 0
	operandsStr := ""
	if len(opDef.OperandWidths) > 0 {
		totalOperandBytes := 0; for _, w := range opDef.OperandWidths { totalOperandBytes += w }
		if ip+1+totalOperandBytes > len(instructions) { return false, fmt.Errorf("%s bounds error reading operands: ip=%d, need %d, len=%d", opDef.Name, ip, totalOperandBytes, len(instructions)) }
		readOperands, numOperandBytes = code.ReadOperands(opDef, instructions[ip+1:])
		if numOperandBytes != totalOperandBytes { return false, fmt.Errorf("failed to read operands for %s at ip=%d", opDef.Name, ip) }
		operandsStr = fmt.Sprintf(" Operands: %v", readOperands)
	}
	fmt.Printf(">>> DEBUG VM: Exec: Frame=%d(%s) IP=%d SP=%d Op=%s%s\n", vm.frameIndex, currentFrame.cl.Fn.Name, ip, vm.sp, opName, operandsStr)

	switch op {
	// --- Constants and Literals ---
	case code.OpConstant:
		constIndex := readOperands[0]
		if int(constIndex) >= len(vm.constants) { execErr = fmt.Errorf("invalid constant index %d (max %d)", constIndex, len(vm.constants)-1); break }
		constantValue := vm.constants[constIndex]
		fmt.Printf("    Constant[%d] = %s\n", constIndex, objInspect(constantValue)); 
		execErr = vm.push(constantValue)
	case code.OpTrue: execErr = vm.push(object.TRUE)
	case code.OpFalse: execErr = vm.push(object.FALSE)
	case code.OpNull: execErr = vm.push(object.NULL)
	// --- Stack Manipulation ---
	case code.OpPop: _, execErr = vm.pop()
	case code.OpDup:
		if vm.sp < 1 { execErr = fmt.Errorf("stack underflow for OpDup"); break }
		top, err := vm.peek(0); if err != nil { execErr = err } else { 
			fmt.Printf("    OpDup: Duplicating %s\n", objInspect(top)); 
			execErr = vm.push(top) 
		}
	// --- Arithmetic, Comparison, Logical, Prefix ---
	case code.OpAdd, code.OpSubtract, code.OpMultiply, code.OpDivide, code.OpModulo: execErr = vm.executeBinaryOperation(op)
	case code.OpEqual, code.OpNotEqual, code.OpGreaterThan, code.OpGreaterThanEqual, code.OpLesserThan, code.OpLesserThanEqual: execErr = vm.executeComparison(op)
	case code.OpBang: execErr = vm.executeBangOperator()
	case code.OpMinus, code.OpPos: execErr = vm.executeUnaryOperator(op)
	// --- Variable Access ---
	case code.OpSetGlobal:
		globalIndex := readOperands[0]
		if vm.sp < 1 { execErr = fmt.Errorf("stack underflow for OpSetGlobal"); break }
		val, err := vm.pop(); if err != nil { execErr = err; break }
		if int(globalIndex) >= len(vm.globals) { execErr = fmt.Errorf("global index %d out of bounds (%d)", globalIndex, len(vm.globals)); break }
		fmt.Printf("    SetGlobal[%d] = %s\n", globalIndex, objInspect(val)); 
		vm.globals[globalIndex] = val
	case code.OpGetGlobal:
		globalIndex := readOperands[0]
		if int(globalIndex) >= len(vm.globals) { execErr = fmt.Errorf("global index %d out of bounds (%d)", globalIndex, len(vm.globals)); break }
		val := vm.globals[globalIndex]; if val == nil { val = object.NULL }
		fmt.Printf("    GetGlobal[%d] = %s\n", globalIndex, objInspect(val)); 
		execErr = vm.push(val)
	case code.OpSetLocal:
		localIndex := readOperands[0]
		if vm.sp < 1 { execErr = fmt.Errorf("stack underflow for OpSetLocal"); break }
		val, err := vm.pop(); if err != nil { execErr = err; break }
		targetSlot := currentFrame.basePointer + 1 + int(localIndex) // Use +1 offset
		if targetSlot < 0 || targetSlot >= StackSize { execErr = fmt.Errorf("invalid target slot %d for OpSetLocal (bp=%d, idx=%d)", targetSlot, currentFrame.basePointer, localIndex); break }
		maxLocals := 0; if currentFrame.cl != nil && currentFrame.cl.Fn != nil { maxLocals = currentFrame.cl.Fn.NumLocals }
        if targetSlot >= currentFrame.basePointer + 1 + maxLocals { execErr = fmt.Errorf("internal error: OpSetLocal target slot %d beyond allocated locals (%d)", targetSlot, currentFrame.basePointer + 1 + maxLocals); break }
		fmt.Printf("    SetLocal[idx=%d, slot=%d] = %s (BP=%d)\n", localIndex, targetSlot, objInspect(val), currentFrame.basePointer); 
		vm.stack[targetSlot] = val
	case code.OpGetLocal:
		localIndex := readOperands[0]
		targetSlot := currentFrame.basePointer + 1 + int(localIndex) // Use +1 offset
		maxLocals := 0; if currentFrame.cl != nil && currentFrame.cl.Fn != nil { maxLocals = currentFrame.cl.Fn.NumLocals }
		if int(localIndex) < 0 || int(localIndex) >= maxLocals { execErr = fmt.Errorf("internal error: accessing invalid local index %d (max %d)", localIndex, maxLocals); break }
		if targetSlot < 0 || targetSlot >= StackSize { execErr = fmt.Errorf("internal error: calculated invalid stack slot %d for local %d (bp=%d)", targetSlot, localIndex, currentFrame.basePointer); break }
		var val object.Object
		// Access slot directly, treat nil as NULL
		val = vm.stack[targetSlot]; if val == nil { val = object.NULL }
		fmt.Printf("    GetLocal[idx=%d, slot=%d] -> %s (BP=%d)\n", localIndex, targetSlot, objInspect(val), currentFrame.basePointer)
		execErr = vm.push(val)
	case code.OpSetFree:
		freeIndex := readOperands[0]
		if vm.sp < 1 { execErr = fmt.Errorf("stack underflow for OpSetFree"); break }
		val, err := vm.pop(); if err != nil { execErr = err; break }
		if currentFrame.cl == nil || int(freeIndex) >= len(currentFrame.cl.Free) { execErr = fmt.Errorf("invalid free var index %d (len=%d)", freeIndex, len(currentFrame.cl.Free)); break }
		fmt.Printf("    SetFree[%d] = %s\n", freeIndex, objInspect(val)); 
		currentFrame.cl.Free[freeIndex] = val
	case code.OpGetFree:
		freeIndex := readOperands[0]
		if currentFrame.cl == nil || int(freeIndex) >= len(currentFrame.cl.Free) { execErr = fmt.Errorf("invalid free var index %d (len=%d)", freeIndex, len(currentFrame.cl.Free)); break }
		val := currentFrame.cl.Free[freeIndex]; if val == nil { val = object.NULL }
		fmt.Printf("    GetFree[%d] = %s\n", freeIndex, objInspect(val)); 
		execErr = vm.push(val)
	case code.OpGetBuiltin:
		builtinIndex := readOperands[0]
		if int(builtinIndex) >= len(vm.builtins) { execErr = fmt.Errorf("invalid builtin index %d", builtinIndex); break }
		builtin := vm.builtins[builtinIndex]
		fmt.Printf("    GetBuiltin[%d] = %s\n", builtinIndex, objInspect(builtin)); 
		execErr = vm.push(builtin)
	// --- Collections ---
	case code.OpArray:
		numElements := readOperands[0]
		if vm.sp < numElements { execErr = fmt.Errorf("stack underflow for OpArray (need %d, have %d)", numElements, vm.sp); break }
		fmt.Printf("    OpArray: Building list with %d elements from SP=%d\n", numElements, vm.sp); 
		elements, err := vm.buildCollection(vm.sp-numElements, vm.sp); if err != nil { execErr = err; break }
		vm.sp -= numElements; listObj := &object.List{Elements: elements}; 
		fmt.Printf("    OpArray: Pushing %s\n", objInspect(listObj)); 
		execErr = vm.push(listObj)
	case code.OpTuple:
		numElements := readOperands[0]
		if vm.sp < numElements { execErr = fmt.Errorf("stack underflow for OpTuple (need %d, have %d)", numElements, vm.sp); break }
		fmt.Printf("    OpTuple: Building tuple with %d elements from SP=%d\n", numElements, vm.sp); 
		elements, err := vm.buildCollection(vm.sp-numElements, vm.sp); if err != nil { execErr = err; break }
		vm.sp -= numElements; tupleObj := &object.Tuple{Elements: elements}; 
		fmt.Printf("    OpTuple: Pushing %s\n", objInspect(tupleObj)); 
		execErr = vm.push(tupleObj)
	case code.OpSet:
		numElements := readOperands[0]
		if vm.sp < numElements { execErr = fmt.Errorf("stack underflow for OpSet (need %d, have %d)", numElements, vm.sp); break }
		fmt.Printf("    OpSet: Building set with %d elements from SP=%d\n", numElements, vm.sp); 
		elementsSlice, err := vm.buildCollection(vm.sp-numElements, vm.sp); if err != nil { execErr = err; break }
		vm.sp -= numElements; set := &object.Set{Elements: make(map[object.HashKey]object.Object, numElements)}
		for _, elem := range elementsSlice { hashableElem, ok := elem.(object.Hashable); if !ok { execErr = object.NewError("TypeError: unhashable type: '%s' in set literal", elem.Type()); break }; hKey, hErr := hashableElem.HashKey(); if hErr != nil { execErr = object.NewError("TypeError: failed to hash element for set: %v", hErr); break }; set.Elements[hKey] = elem }
		if execErr != nil { break }; 
		fmt.Printf("    OpSet: Pushing %s\n", objInspect(set)); 
		execErr = vm.push(set)
	case code.OpHash:
		numPairs := readOperands[0]; itemsNeeded := numPairs * 2
		if vm.sp < itemsNeeded { execErr = fmt.Errorf("stack underflow for OpHash (need %d, have %d)", itemsNeeded, vm.sp); break }
		fmt.Printf("    OpHash: Building dict with %d pairs from SP=%d\n", numPairs, vm.sp); 
		startIndex := vm.sp - itemsNeeded; 
		if startIndex < 0 { execErr = fmt.Errorf("internal error: invalid startIndex building hash"); break }
		pairs := make(map[object.HashKey]object.DictPair, numPairs)
		for i := startIndex; i < vm.sp; i += 2 { key := vm.stack[i]; value := vm.stack[i+1]; hashableKey, ok := key.(object.Hashable); if !ok { execErr = object.NewError("TypeError: unhashable type: '%s' as dict key", key.Type()); break }; hashKeyVal, err := hashableKey.HashKey(); if err != nil { execErr = object.NewError("TypeError: failed to hash key: %v", err); break }; pairs[hashKeyVal] = object.DictPair{Key: key, Value: value} }
		if execErr != nil { break }; for i := startIndex; i < vm.sp; i++ { vm.stack[i] = nil }; vm.sp = startIndex; dictObj := &object.Dict{Pairs: pairs}; 
		fmt.Printf("    OpHash: Pushing %s\n", objInspect(dictObj)); 
		execErr = vm.push(dictObj)
	case code.OpIndex:
		if vm.sp < 2 { execErr = fmt.Errorf("stack underflow for OpIndex"); break }
		index, err := vm.pop(); if err != nil { execErr = err; break }; left, err := vm.pop(); if err != nil { execErr = err; break }
		fmt.Printf("    OpIndex: Collection=%s, Index=%s\n", objInspect(left), objInspect(index)); 
		execErr = vm.executeIndexOperation(left, index)
	case code.OpSetIndex:
		if vm.sp < 3 { execErr = fmt.Errorf("stack underflow for OpSetIndex"); break }
		value, err := vm.pop(); if err != nil { execErr = err; break }; index, err := vm.pop(); if err != nil { execErr = err; break }; collection, err := vm.pop(); if err != nil { execErr = err; break }
		fmt.Printf("    OpSetIndex: Collection=%s, Index=%s, Value=%s\n", objInspect(collection), objInspect(index), objInspect(value)); 
		execErr = vm.executeIndexAssignment(collection, index, value)
	case code.OpGetIter:
		if vm.sp < 1 { execErr = fmt.Errorf("stack underflow for OpGetIter"); break }
		iterable, err := vm.pop(); if err != nil { execErr = err; break }
		fmt.Printf("    OpGetIter: Getting iterator for %s\n", objInspect(iterable)); 
		iterator, errIter := vm.getObjectIterator(iterable); if errIter != nil { execErr = errIter; break }
		fmt.Printf("    OpGetIter: Pushing iterator %s\n", objInspect(iterator)); 
		execErr = vm.push(iterator)
	// --- Attributes and Methods ---
	case code.OpGetAttribute:
		nameConstIdx := readOperands[0]
		if vm.sp < 1 { execErr = fmt.Errorf("stack underflow for OpGetAttribute object"); break }
		nameObj, ok := vm.constants[nameConstIdx].(*object.String); if !ok { execErr = fmt.Errorf("invalid attr name constant %d (not String)", nameConstIdx); break }
		obj, err := vm.pop(); if err != nil { execErr = err; break }
		fmt.Printf("    OpGetAttribute: Object=%s, AttrName=%q\n", objInspect(obj), nameObj.Value); 
		execErr = vm.executeGetAttribute(obj, nameObj.Value)
	case code.OpSetAttribute:
		nameConstIdx := readOperands[0]
		if vm.sp < 2 { execErr = fmt.Errorf("stack underflow for OpSetAttribute value/object"); break }
		nameObj, ok := vm.constants[nameConstIdx].(*object.String); if !ok { execErr = fmt.Errorf("invalid attr name constant %d (not String)", nameConstIdx); break }
		value, err := vm.pop(); if err != nil { execErr = err; break }; obj, err := vm.pop(); if err != nil { execErr = err; break }
		fmt.Printf("    OpSetAttribute: Object=%s, AttrName=%q, Value=%s\n", objInspect(obj), nameObj.Value, objInspect(value)); 
		execErr = vm.executeSetAttribute(obj, nameObj.Value, value)
	// --- Functions, Closures, Classes ---
	case code.OpClosure:
		constIndex := readOperands[0]; numFree := readOperands[1]
		compiledFn, ok := vm.constants[constIndex].(*core.CompiledFunction); if !ok { execErr = fmt.Errorf("constant %d is not a CompiledFunction", constIndex); break }
		fmt.Printf("    OpClosure: Reading %d free vars (SP before pop: %d)\n", numFree, vm.sp); 
		if vm.sp < int(numFree) { execErr = fmt.Errorf("stack underflow popping free vars for closure (need %d, have %d)", numFree, vm.sp); break }
		freeVars := make([]object.Object, numFree)
		for i := int(numFree) - 1; i >= 0; i-- { freeVars[i], execErr = vm.pop(); if execErr != nil { execErr = fmt.Errorf("OpClosure pop free var error (index %d): %w", i, execErr); break } ; 
		fmt.Printf("    OpClosure: Popped free var %d: %s\n", i, objInspect(freeVars[i])) }
		if execErr != nil { break }; closure := &Closure{Fn: compiledFn, Free: freeVars}; 
		fmt.Printf("    OpClosure: Pushing %s\n", objInspect(closure)); 
		execErr = vm.push(closure)
	case code.OpBuildClass:
		numMethods := readOperands[0]; itemsToPop := numMethods*2 + 1
		if vm.sp < itemsToPop { execErr = fmt.Errorf("OpBuildClass stack underflow (need %d items, have %d)", itemsToPop, vm.sp); break }
		fmt.Printf("    OpBuildClass: Building class with %d methods from SP=%d\n", numMethods, vm.sp); 
		methods := make(map[string]*Closure, numMethods)
		for i := 0; i < numMethods; i++ { closureObj, err := vm.pop(); if err != nil { execErr=fmt.Errorf("OpBuildClass pop closure err: %w", err); break }; closure, ok := closureObj.(*Closure); if !ok { execErr = fmt.Errorf("OpBuildClass expected Closure, got %s", closureObj.Type()); break }; nameObj, err := vm.pop(); if err != nil { execErr=fmt.Errorf("OpBuildClass pop name err: %w", err); break }; nameStr, ok := nameObj.(*object.String); if !ok { execErr = fmt.Errorf("OpBuildClass expected String name, got %s", nameObj.Type()); break }; methods[nameStr.Value] = closure }
		if execErr != nil { break }; classNameObj, err := vm.pop(); if err != nil { execErr=fmt.Errorf("OpBuildClass pop class name err: %w", err); break }; classNameStr, ok := classNameObj.(*object.String); if !ok { execErr = fmt.Errorf("OpBuildClass expected String class name, got %s", classNameObj.Type()); break }; classObj := &Class{Name: classNameStr.Value, Methods: methods}; 
		fmt.Printf("    OpBuildClass: Popped ClassName=%q. Pushing %s\n", classNameStr.Value, objInspect(classObj)); 
		execErr = vm.push(classObj)
	// --- Module Import ---
	case code.OpImportName:
		nameIndex := readOperands[0]
		nameObj, ok := vm.constants[nameIndex].(*object.String); if !ok { execErr = fmt.Errorf("invalid module name constant %d (not String)", nameIndex); break }
		moduleName := nameObj.Value; 
		fmt.Printf("    OpImportName: Importing '%s'\n", moduleName); 
		module, found := vm.nativeModules[moduleName]; if !found { execErr = object.NewError("ImportError: no built-in module named '%s'", moduleName); break }
		fmt.Printf("    OpImportName: Pushing module %s\n", objInspect(module)); 
		execErr = vm.push(module)
	// --- Exception Handling ---
	case code.OpPushExceptionHandler:
		exceptTarget := readOperands[0]; finallyTarget := readOperands[1]
		if exceptTarget < 0 || exceptTarget >= len(instructions) { execErr = fmt.Errorf("invalid except target %d in OpPushExceptionHandler", exceptTarget); break }
		if finallyTarget != 0 && (finallyTarget < 0 || finallyTarget >= len(instructions)) { execErr = fmt.Errorf("invalid finally target %d in OpPushExceptionHandler", finallyTarget); break }
		handler := &ExceptionHandler{ ExceptTargetIP: exceptTarget, FinallyTargetIP: finallyTarget, StackPointerLevel: vm.sp, FrameIndex: vm.frameIndex }; 
		fmt.Printf("    OpPushExceptionHandler: ExceptIP=%d, FinallyIP=%d, SPLevel=%d, Frame=%d\n", exceptTarget, finallyTarget, vm.sp, vm.frameIndex); 
		vm.handlerStack = append(vm.handlerStack, handler)
	case code.OpPopExceptionHandler:
		if len(vm.handlerStack) == 0 { execErr = fmt.Errorf("internal VM error: OpPopExceptionHandler with empty handler stack"); break }
		fmt.Printf("    OpPopExceptionHandler: Popping handler\n"); 
		vm.handlerStack = vm.handlerStack[:len(vm.handlerStack)-1]
	case code.OpSwap:
		if vm.sp < 2 { execErr = fmt.Errorf("stack underflow for OpSwap"); break }
		obj1, err := vm.pop(); if err != nil { execErr = err; break }; obj2, err := vm.pop(); if err != nil { _ = vm.push(obj1); execErr = err; break }
		fmt.Printf("    OpSwap: Swapping %s and %s\n", objInspect(obj1), objInspect(obj2)); 
		err = vm.push(obj1); if err != nil { execErr = err; break }; execErr = vm.push(obj2)
	case code.OpIsInstance:
		if vm.sp < 2 { execErr = fmt.Errorf("stack underflow for OpIsInstance"); break }
		classInfo, err := vm.pop(); if err != nil { execErr = err; break }; obj, err := vm.pop(); if err != nil { execErr = err; break }
		fmt.Printf("    OpIsInstance: Checking if %s is instance of %s\n", objInspect(obj), objInspect(classInfo)); 
		resultBool, err := vm.isInstance(obj, classInfo); if err != nil { execErr = err; break }
		fmt.Printf("      Result: %t\n", resultBool); 
		if resultBool { err = vm.push(object.TRUE) } else { err = vm.push(object.FALSE) }; execErr = err
	case code.OpRaise:
		if vm.sp < 1 { execErr = fmt.Errorf("stack underflow for OpRaise"); break }
		exceptionObj, err := vm.pop(); if err != nil { execErr = err; break }
		fmt.Printf("    OpRaise: Raising %s\n", objInspect(exceptionObj)); 
		if errInstance, ok := exceptionObj.(*object.Error); ok { execErr = errInstance } else if stopIterInstance, ok2 := exceptionObj.(*object.StopIterationError); ok2 { execErr = stopIterInstance } else { execErr = object.NewError("TypeError: exceptions must derive from BaseException, not %s", exceptionObj.Type()) }

	// --- Jumps (Modify IP directly, return ipWasSet=true) ---
	case code.OpJump:
		jumpPos := readOperands[0]
		if jumpPos < 0 || jumpPos >= len(instructions) { return true, fmt.Errorf("OpJump target %d out of bounds [0, %d)", jumpPos, len(instructions)) }
		fmt.Printf("    OpJump: Jumping from IP %d to IP %d\n", ip, jumpPos); 
		currentFrame.ip = jumpPos - 1 // Set IP for *next* cycle calculation
		ipWasSetByJump = true; return ipWasSetByJump, nil
	case code.OpJumpNotTruthy:
		jumpPos := readOperands[0]
		if jumpPos < 0 || jumpPos >= len(instructions) { return false, fmt.Errorf("OpJumpNotTruthy target %d out of bounds [0, %d)", jumpPos, len(instructions)) }
		if vm.sp < 1 { return false, fmt.Errorf("stack underflow for OpJumpNotTruthy condition") }
		condition, err := vm.pop(); if err != nil { return false, err }
		isTrue := vm.isTruthy(condition); 
		fmt.Printf("    OpJumpNotTruthy: Condition=%s (%t), Target IP=%d\n", objInspect(condition), !isTrue, jumpPos)
		if !isTrue { 
			fmt.Printf("        Jumping to %d\n", jumpPos); 
			currentFrame.ip = jumpPos - 1; 
			ipWasSetByJump = true; 
			return ipWasSetByJump, nil 
		}
		// else: Fall through, ipWasSetByJump remains false
	case code.OpJumpIfTruthy:
		jumpPos := readOperands[0]
		if jumpPos < 0 || jumpPos >= len(instructions) { return false, fmt.Errorf("OpJumpIfTruthy target %d out of bounds [0, %d)", jumpPos, len(instructions)) }
		if vm.sp < 1 { return false, fmt.Errorf("stack underflow for OpJumpIfTruthy condition") }
		condition, err := vm.pop(); if err != nil { return false, err }
		isTrue := vm.isTruthy(condition); 
		fmt.Printf("    OpJumpIfTruthy: Condition=%s (%t), Target IP=%d\n", objInspect(condition), isTrue, jumpPos)
		if isTrue { 
			fmt.Printf("        Jumping to %d\n", jumpPos); 
			currentFrame.ip = jumpPos - 1; 
			ipWasSetByJump = true; 
			return ipWasSetByJump, nil 
		}
		// else: Fall through, ipWasSetByJump remains false
	case code.OpForIter:
		jumpPos := readOperands[0]
		if jumpPos < 0 || jumpPos >= len(instructions) { return false, fmt.Errorf("OpForIter target %d out of bounds [0, %d)", jumpPos, len(instructions)) }
		if vm.sp < 1 { return false, fmt.Errorf("stack underflow: OpForIter requires iterator") }
		iterObj, err := vm.peek(0); if err != nil { return false, err }
		iterator, ok := iterObj.(object.Iterator); if !ok { return false, fmt.Errorf("internal VM error: OpForIter expected iterator, got %s", iterObj.Type()) }
		nextVal, stop := iterator.Next(); 
		fmt.Printf("    OpForIter: Next() -> %s, Stop=%t, JumpTarget=%d\n", objInspect(nextVal), stop, jumpPos)
		if stop { _, err = vm.pop(); if err != nil { return false, err }; 
		fmt.Printf("        Iterator finished, jumping to %d\n", jumpPos); 
		currentFrame.ip = jumpPos - 1; ipWasSetByJump = true; return ipWasSetByJump, nil }
		err = vm.push(nextVal); if err != nil { return false, err }
		// else: Fall through (pushed value), ipWasSetByJump remains false

	// --- Calls & Returns (Frame change handles flow, DO NOT set ipWasSetByJump) ---
	case code.OpCall:
		numArgs := readOperands[0]; requiredOnStack := numArgs + 1
		if vm.sp < requiredOnStack { return false, fmt.Errorf("stack underflow preparing for OpCall (need %d, have %d)", requiredOnStack, vm.sp) }
		callable, err := vm.peek(numArgs); if err != nil { return false, err }
		fmt.Printf("    OpCall: numArgs=%d, Callable=%s\n", numArgs, objInspect(callable))
		switch callee := callable.(type) {
		case *Closure: execErr = vm.callClosure(callee, numArgs)
		case *object.Builtin: execErr = vm.callBuiltin(callee, numArgs)
		case *Class: execErr = vm.callClass(callee, numArgs)
		case *BoundMethod: execErr = vm.callBoundMethod(callee, numArgs)
		default: execErr = object.NewError("TypeError: '%s' object is not callable", callee.Type())
		}
		// Run loop handles IP based on frame change or lack thereof (for builtins)
		return false, execErr // ipWasSetByJump is false
	case code.OpReturnValue:
		if vm.sp < 1 { return false, fmt.Errorf("stack underflow for OpReturnValue") }
		returnValue, err := vm.pop(); if err != nil { return false, err }
		if vm.frameIndex < 0 { return false, fmt.Errorf("internal error: OpReturnValue with no frame to pop") }
		poppedFrame := vm.currentFrame() // Get frame *before* popping for context
		basePointer := poppedFrame.basePointer; if basePointer < 0 { return false, fmt.Errorf("internal error: invalid basepointer %d for return", basePointer)}
        finalValue := returnValue
        // Handle __init__ return None -> self
		if poppedFrame.cl != nil && poppedFrame.cl.Fn != nil && poppedFrame.cl.Fn.Name == "__init__" {
            selfSlot := basePointer + 1; // __init__'s first arg is self at BP+1
            if selfSlot >= 0 && selfSlot < StackSize { selfObj := vm.stack[selfSlot]; if inst, ok := selfObj.(*Instance); ok { if returnValue != object.NULL { execErr = object.NewError("TypeError: __init__() should return None, not %s", returnValue.Type()) } else { finalValue = inst; 
				fmt.Printf("      Overriding return with 'self' instance: %s\n", objInspect(finalValue)) 
				} 
			} else { execErr = fmt.Errorf("internal VM error: __init__ frame expected Instance at slot %d, found %T", selfSlot, selfObj) } } else { execErr = fmt.Errorf("internal VM error: __init__ frame invalid self slot %d", selfSlot) }
            if execErr != nil { return false, execErr } // Return __init__ specific error
		}
        // Place return value at base pointer slot (overwriting func/closure)
        if basePointer >= StackSize { return false, fmt.Errorf("internal error: basepointer %d out of bounds for return value", basePointer)}
        vm.stack[basePointer] = finalValue
        // Reset SP to point just AFTER the return value
        vm.sp = basePointer + 1
		fmt.Printf("    OpReturnValue: Prepared return value %s at slot %d. Signaling return.\n", objInspect(finalValue), basePointer)
		return false, ErrReturnSignal // Use sentinel error, ipWasSetByJump is false
	case code.OpReturn: // Void return
		if vm.frameIndex < 0 { return false, fmt.Errorf("internal error: OpReturn with no frame to pop") }
		poppedFrame := vm.currentFrame() // Get frame *before* popping
		basePointer := poppedFrame.basePointer; if basePointer < 0 { return false, fmt.Errorf("internal error: invalid basepointer %d for return", basePointer)}
		var finalValue object.Object = object.NULL
        // Handle __init__ return None -> self
		if poppedFrame.cl != nil && poppedFrame.cl.Fn != nil && poppedFrame.cl.Fn.Name == "__init__" {
            selfSlot := basePointer + 1; // __init__'s first arg is self at BP+1
            if selfSlot >= 0 && selfSlot < StackSize { selfObj := vm.stack[selfSlot]; if inst, ok := selfObj.(*Instance); ok { finalValue = inst; 
				fmt.Printf("      Overriding return with 'self' instance: %s\n", objInspect(finalValue)) 
			} else { execErr = fmt.Errorf("internal VM error: __init__ frame expected Instance at slot %d, found %T", selfSlot, selfObj) } } else { execErr = fmt.Errorf("internal VM error: __init__ frame invalid self slot %d", selfSlot) }
            if execErr != nil { return false, execErr } // Return __init__ specific error
		}
        // Place NULL/self at base pointer slot
        if basePointer >= StackSize { return false, fmt.Errorf("internal error: basepointer %d out of bounds for return value", basePointer)}
        vm.stack[basePointer] = finalValue
        // Reset SP to point just AFTER the return value
        vm.sp = basePointer + 1
		fmt.Printf("    OpReturn: Prepared return value %s at slot %d. Signaling void return.\n", objInspect(finalValue), basePointer)
		return false, ErrVoidReturnSignal // Use sentinel error, ipWasSetByJump is false

	default:
		execErr = fmt.Errorf("unknown/unhandled %s executed", opName)
	} // End Opcode Switch

	// Return ipWasSetByJump=false and any error for normal instructions that fall through
	return ipWasSetByJump, execErr
}



// --- Run Loop (Main VM Execution - FINAL VERSION with Post-Return Fix) ---
func (vm *VM) Run() (runErr error) {
	// Defer panic recovery for the main run loop
	defer func() {
		if r := recover(); r != nil {
			// Create a Pylearn error object from the panic
			runErr = object.NewError("Panic recovered in VM: %v", r)
			fmt.Printf(">>> DEBUG VM: PANIC RECOVERED in Run loop: %v\n", r)
			// Ensure termination state
			vm.frameIndex = -1
		}
	}()

runLoop:
	for {
		// --- 1. Termination Check ---
		if vm.frameIndex < 0 {
			// ___fmt.Println(">>> DEBUG VM:   Terminating Run loop (frameIndex < 0)")
			break runLoop
		}

		// --- 2. Callback Check ---
		select {
		case req := <-vm.callbackChan:
			// ___fmt.Println(">>> DEBUG VM:   Handling callback request in Run loop.")
			callbackResult := vm.executeCallback(req) // Executes synchronously
			req.ResultChan <- callbackResult          // Send result back
			// After handling a callback, the frame index or state might have changed.
			// Restarting the loop ensures we re-evaluate the current state.
			continue runLoop
		default:
			// No callback waiting, proceed with instruction execution
		}

		// --- 3. Get Current State ---
		currentFrame := vm.currentFrame() // Panics if frameIndex is invalid
		instructions := currentFrame.Instructions()

		// --- 4. Calculate IP for this cycle's execution ---
		// currentFrame.ip holds the IP of the *last successfully completed* instruction's opcode,
		// or -1 if the frame just started.
		ipToExecute := currentFrame.ip + 1

		// --- 5. Check IP Bounds ---
		if ipToExecute < 0 || ipToExecute >= len(instructions) {
			fmt.Printf(">>> DEBUG VM: IP %d is out of bounds [0, %d) for Frame %d (%s). Checking termination...\n",
				ipToExecute, len(instructions), vm.frameIndex, currentFrame.cl.Fn.Name)
			// Check for normal end of main script vs. error
			if vm.frameIndex == 0 && ipToExecute == len(instructions) {
				// ___fmt.Println(">>> DEBUG VM:   Reached end of main script instructions normally.")
			} else if ipToExecute < 0 {
				runErr = fmt.Errorf("internal vm error: negative IP calculated (%d)", ipToExecute)
				fmt.Printf(">>> DEBUG VM: Error: %v. Forcing termination.\n", runErr)
			} else {
				runErr = fmt.Errorf("internal vm error: IP %d out of bounds [0, %d) in frame %d (%s)",
					ipToExecute, len(instructions), vm.frameIndex, currentFrame.cl.Fn.Name)
				fmt.Printf(">>> DEBUG VM: Error: %v. Forcing termination.\n", runErr)
			}
			vm.frameIndex = -1 // Force termination
			break runLoop
		}

		// --- 6. Commit IP for this cycle ---
		currentFrame.ip = ipToExecute

		// --- 7. Decode Opcode (for error reporting and width calculation) ---
		op := code.Opcode(instructions[ipToExecute])
		opDef, err := code.Lookup(byte(op))
		if err != nil {
			runErr = fmt.Errorf("internal vm error: unknown opcode %d at ip %d in frame %d (%s)",
				op, ipToExecute, vm.frameIndex, currentFrame.cl.Fn.Name)
			fmt.Printf(">>> DEBUG VM: Error: %v. Forcing termination.\n", runErr)
			vm.frameIndex = -1
			break runLoop
		}

		// --- Store state BEFORE execution might change it ---
		frameIdxBefore := vm.frameIndex
		ipBeforeExecute := ipToExecute // IP of the instruction *being* executed now

		// --- 8. Execute Instruction ---
		// ipWasSetByJump is true ONLY if a jump occurred within executeInstruction
		ipWasSetByJump, execErr := vm.executeInstruction(currentFrame) // Uses currentFrame.ip, ONLY sets it for jumps

		// --- 9. Handle Return Signals FIRST ---
		// isReturnSignal := false
		if execErr != nil {
			if errors.Is(execErr, ErrReturnSignal) || errors.Is(execErr, ErrVoidReturnSignal) {
				// isReturnSignal = true
				fmt.Printf(">>> DEBUG VM: Return signal received from frame %d\n", frameIdxBefore)

				// Pop the frame THAT signaled return.
				// The return value is already placed on the stack at the correct base pointer slot.
				vm.popFrame()

				if vm.frameIndex < 0 { // Check if we returned from the main script
					// ___fmt.Println(">>> DEBUG VM:   Return popped last frame.")
					runErr = nil // Normal termination
					break runLoop
				}

				// --- Adjust IP in the CALLER frame ---
				callerFrame := vm.currentFrame() // Frame we returned TO

				// ipBeforeExecute holds the IP of the OpCall instruction in the *caller* frame.
				opCallIP := ipBeforeExecute

				// We need the definition of the OpCall instruction at that IP in the caller.
				callerInstructions := callerFrame.Instructions()
				if opCallIP < 0 || opCallIP >= len(callerInstructions) {
					runErr = fmt.Errorf("internal error: invalid OpCall IP %d recorded before return", opCallIP)
					vm.frameIndex = -1
					break runLoop
				}
				opCallOpcodeByte := callerInstructions[opCallIP]
				callOpDef, lookupErr := code.Lookup(opCallOpcodeByte)

				// Verify it was indeed an OpCall and calculate its full width
				if lookupErr == nil && callOpDef.Name == "OpCall" {
					callInstructionWidth := 1 // opcode byte
					for _, w := range callOpDef.OperandWidths {
						callInstructionWidth += w
					}
					// Set IP to the *last byte* of the OpCall instruction in the caller frame.
					callerFrame.ip = opCallIP + callInstructionWidth - 1
					fmt.Printf(">>> DEBUG VM: Post-Return IP Adjust: Caller IP set to %d (after OpCall at %d)\n", callerFrame.ip, opCallIP)
				} else {
					// If it wasn't OpCall, log a warning. The IP remains pointing at the
					// OpCall opcode (ipBeforeExecute). The next loop iteration should advance past it.
					fmt.Printf(">>> DEBUG VM: WARNING - Return signal but instruction at caller IP %d wasn't OpCall? Found %d (%v). Leaving IP as %d.\n",
						opCallIP, opCallOpcodeByte, lookupErr, callerFrame.ip)
                    // Let the default IP advancement handle this unexpected case if possible.
                    // This relies on the next loop iteration advancing past the (potentially non-OpCall) instruction.
				}

				execErr = nil     // Clear the signal error
				continue runLoop // Important: Skip default IP advance after return adjustment
			}
		}

		// --- 10. Handle REAL Execution Errors ---
		if execErr != nil {
			fmt.Printf(">>> DEBUG VM: Instruction %s at IP=%d resulted in error: %v. Handling...\n", opDef.Name, ipBeforeExecute, execErr)
			var pyErrInstance error; if errObj, ok := execErr.(error); ok { _, isObjErr := execErr.(*object.Error); _, isStopIterErr := execErr.(*object.StopIterationError); if isObjErr || isStopIterErr { pyErrInstance = errObj } else { pyErrInstance = object.NewError("RuntimeError: %v", execErr) } } else { pyErrInstance = object.NewError("InternalError: Unknown non-error type returned: %T value: %v", execErr, execErr) }
			handlerFound := false
			for len(vm.handlerStack) > 0 {
				handler := vm.handlerStack[len(vm.handlerStack)-1]; if handler.FrameIndex > vm.frameIndex { runErr = object.NewError("Internal VM Error: Handler frame index %d > current frame index %d", handler.FrameIndex, vm.frameIndex); vm.frameIndex = -1; break runLoop }
				vm.handlerStack = vm.handlerStack[:len(vm.handlerStack)-1]; handlerFound = true; 
				fmt.Printf(">>> DEBUG VM: Found handler targeting frame %d, exceptIP=%d\n", handler.FrameIndex, handler.ExceptTargetIP)
				for vm.frameIndex > handler.FrameIndex { if vm.frameIndex < 0 { runErr = object.NewError("Internal VM Error: Frame index < 0 during exception unwind."); break runLoop }; 
				fmt.Printf(">>> DEBUG VM: Unwinding frame %d for exception\n", vm.frameIndex); vm.popFrame() 
			}
				if vm.frameIndex != handler.FrameIndex { runErr = object.NewError("Internal VM Error: Frame index mismatch after unwind (expected %d, got %d)", handler.FrameIndex, vm.frameIndex); break runLoop }
				currentFrame = vm.currentFrame() // Update currentFrame after potential pops
				if handler.StackPointerLevel < 0 || handler.StackPointerLevel > vm.sp { runErr = object.NewError("Internal VM Error: Invalid SP level (%d vs current %d) in handler", handler.StackPointerLevel, vm.sp); break runLoop }
				fmt.Printf(">>> DEBUG VM: Unwinding stack from SP %d to %d\n", vm.sp, handler.StackPointerLevel); for i := handler.StackPointerLevel; i < vm.sp; i++ { vm.stack[i] = nil }; vm.sp = handler.StackPointerLevel
				var pyErrorObject object.Object; if objErr, ok := pyErrInstance.(object.Object); ok { pyErrorObject = objErr } else { pyErrorObject = object.NewError("Internal VM Error: Cannot push non-object error: %v", pyErrInstance) }
				errPush := vm.push(pyErrorObject); if errPush != nil { runErr = object.NewError("Internal VM Error: Stack overflow pushing exception object: %v", errPush); break runLoop }
				fmt.Printf(">>> DEBUG VM: Pushed exception %s, Jumping to handler IP %d\n", objInspect(pyErrorObject), handler.ExceptTargetIP)
				targetIP := handler.ExceptTargetIP; if targetIP < 0 || targetIP >= len(currentFrame.Instructions()) { runErr = object.NewError("Internal VM Error: Invalid ExceptTargetIP %d in handler", targetIP); break runLoop }
				currentFrame.ip = targetIP - 1 // Set IP for next cycle calculation
				continue runLoop // Continue execution in the exception handler block
			}
			if !handlerFound { 
				// ___fmt.Println(">>> DEBUG VM:   No handler found, terminating run loop with error."); 
				runErr = pyErrInstance; 
				vm.frameIndex = -1; 
				break runLoop 
			}
		} // End Error Handling

		// --- 11. Advance IP for NEXT cycle (if not handled by jump/return signal) ---
		// Note: isReturnSignal check is implicit because we 'continue runLoop' if it was true
		if !ipWasSetByJump {
			// Normal instruction execution, advance past operands
			operandWidth := 0
			if opDef != nil { // opDef lookup succeeded earlier
				for _, w := range opDef.OperandWidths { operandWidth += w }
			}
			// currentFrame.ip currently holds ipToExecute (the opcode IP)
			currentFrame.ip += operandWidth // Advance past executed instruction operands
			fmt.Printf(">>> DEBUG VM: Normal Advance: IP set to %d (after %s at %d)\n", currentFrame.ip, opDef.Name, ipBeforeExecute) // Optional log
		}
		// else: IP was set by jump, or control flow changed via return (handled above). The IP is already correct for the next iteration.

	} // End VM run loop

	fmt.Printf(">>> DEBUG VM: Run loop finished outside loop. Final FrameIdx=%d, Final SP=%d, Error=%v\n", vm.frameIndex, vm.sp, runErr)
	return runErr
}

func (vm *VM) executeBinaryOperation(op code.Opcode) error {
	right, err := vm.pop()
	if err != nil { return err } // Propagate Go errors (stack underflow)
	left, err := vm.pop()
	if err != nil { return err }

	leftType := left.Type()
	rightType := right.Type()

	if leftType == object.INTEGER_OBJ && rightType == object.INTEGER_OBJ {
		return vm.executeBinaryIntegerOperation(op, left, right) // Returns error (Go or Py)
	}
	if object.IsNumeric(left) && object.IsNumeric(right) {
		lF, rF := object.PromoteToFloats(left, right)
		return vm.executeBinaryFloatOperation(op, lF, rF) // Returns error (Go or Py)
	}
	if leftType == object.STRING_OBJ && rightType == object.STRING_OBJ {
		return vm.executeBinaryStringOperation(op, left, right) // Returns error (Go or Py)
	}

	// Default: Unsupported types
	return object.NewError("TypeError: unsupported operand type(s) for %s: '%s' and '%s'", op, leftType, rightType)
}

func (vm *VM) executeBinaryIntegerOperation(op code.Opcode, left, right object.Object) error {
	lVal := left.(*object.Integer).Value
	rVal := right.(*object.Integer).Value
	var result int64
	switch op {
	case code.OpAdd:
		result = lVal + rVal
	case code.OpSubtract:
		result = lVal - rVal
	case code.OpMultiply:
		result = lVal * rVal
	case code.OpDivide:
		if rVal == 0 {
			return object.NewError("ZeroDivisionError: division by zero") // Py Error
		}
		// Use push directly, return its error if stack overflows
		return vm.push(&object.Float{Value: float64(lVal) / float64(rVal)})
	case code.OpModulo:
		if rVal == 0 {
			return object.NewError("ZeroDivisionError: integer modulo by zero") // Py Error
		}
		// Use push directly, return its error if stack overflows
		return vm.push(&object.Integer{Value: lVal % rVal})
	default:
		return fmt.Errorf("unknown integer binary operator: %s", op) // Internal Go Error
	}
	// Use push for results, return its error if stack overflows
	return vm.push(&object.Integer{Value: result})
}

func (vm *VM) executeBinaryFloatOperation(op code.Opcode, left, right *object.Float) error {
	lVal := left.Value
	rVal := right.Value
	var result float64
	switch op {
	case code.OpAdd:
		result = lVal + rVal
	case code.OpSubtract:
		result = lVal - rVal
	case code.OpMultiply:
		result = lVal * rVal
	case code.OpDivide:
		if rVal == 0.0 {
			return object.NewError("ZeroDivisionError: float division by zero") // Py Error
		}
		result = lVal / rVal
	case code.OpModulo:
		return object.NewError("NotImplementedError: float modulo not implemented")
	default:
		return fmt.Errorf("unknown float binary operator: %s", op) // Internal Go Error
	}
	return vm.push(&object.Float{Value: result}) // Push float result
}

func (vm *VM) executeBinaryStringOperation(op code.Opcode, left, right object.Object) error {
	lVal := left.(*object.String).Value
	rVal := right.(*object.String).Value
	if op == code.OpAdd {
		return vm.push(&object.String{Value: lVal + rVal})
	}
	return object.NewError("TypeError: unsupported operand type(s) for %s: 'str' and 'str'", op)
}

// --- Comparison Ops ---
func (vm *VM) executeComparison(op code.Opcode) error {
	right, err := vm.pop(); if err != nil { return err }
	left, err := vm.pop(); if err != nil { return err }

	result := false
	// Numeric Comparison
	if object.IsNumeric(left) && object.IsNumeric(right) {
		lF, rF := object.PromoteToFloats(left, right)
		switch op {
		case code.OpEqual: result = lF.Value == rF.Value
		case code.OpNotEqual: result = lF.Value != rF.Value
		case code.OpGreaterThan: result = lF.Value > rF.Value
		case code.OpGreaterThanEqual: result = lF.Value >= rF.Value
		case code.OpLesserThan: result = lF.Value < rF.Value
		case code.OpLesserThanEqual: result = lF.Value <= rF.Value
		default: return fmt.Errorf("unknown numeric comparison: %s", op)
		}
	} else if left.Type() == object.BOOLEAN_OBJ && right.Type() == object.BOOLEAN_OBJ {
		lVal, rVal := left.(*object.Boolean).Value, right.(*object.Boolean).Value
		switch op {
		case code.OpEqual: result = (lVal == rVal)
		case code.OpNotEqual: result = (lVal != rVal)
		default: return object.NewError("TypeError: '%s' not supported between instances of 'bool' and 'bool'", op)
		}
	} else if left.Type() == object.STRING_OBJ && right.Type() == object.STRING_OBJ {
		lVal, rVal := left.(*object.String).Value, right.(*object.String).Value
		switch op {
		case code.OpEqual: result = lVal == rVal
		case code.OpNotEqual: result = lVal != rVal
		case code.OpGreaterThan: result = lVal > rVal
		case code.OpGreaterThanEqual: result = lVal >= rVal
		case code.OpLesserThan: result = lVal < rVal
		case code.OpLesserThanEqual: result = lVal <= rVal
		default: return fmt.Errorf("unknown string comparison: %s", op)
		}
	} else if left == object.NULL || right == object.NULL {
		switch op {
		case code.OpEqual: result = (left == right)
		case code.OpNotEqual: result = (left != right)
		default: return object.NewError("TypeError: '%s' not supported between instances of '%s' and 'NoneType'", op, left.Type())
		}
	} else { // Default comparison (identity for ==/!=, TypeError otherwise)
		switch op {
		case code.OpEqual: result = (left == right)
		case code.OpNotEqual: result = (left != right)
		default: return object.NewError("TypeError: '%s' not supported between instances of '%s' and '%s'", op, left.Type(), right.Type())
		}
	}
	if result { return vm.push(object.TRUE) }
	return vm.push(object.FALSE)
}

// --- Unary Ops ---
func (vm *VM) executeBangOperator() error {
	operand, err := vm.pop(); if err != nil { return err }
	if vm.isTruthy(operand) { return vm.push(object.FALSE) }
	return vm.push(object.TRUE)
}

func (vm *VM) executeUnaryOperator(op code.Opcode) error {
	operand, err := vm.pop(); if err != nil { return err }
	switch op {
	case code.OpMinus:
		switch opType := operand.(type) {
		case *object.Integer: return vm.push(&object.Integer{Value: -opType.Value})
		case *object.Float: return vm.push(&object.Float{Value: -opType.Value})
		default: return object.NewError("TypeError: bad operand type for unary -: '%s'", operand.Type())
		}
	case code.OpPos:
		switch operand.Type() {
		case object.INTEGER_OBJ, object.FLOAT_OBJ: return vm.push(operand)
		default: return object.NewError("TypeError: bad operand type for unary +: '%s'", operand.Type())
		}
	}
	return fmt.Errorf("internal vm error: unhandled unary operator %s", op)
}

// --- Truthiness ---
func (vm *VM) isTruthy(obj object.Object) bool { // Simplified version for VM
	switch obj := obj.(type) {
	case *object.Boolean: return obj.Value
	case *object.Null: return false
	case *object.Integer: return obj.Value != 0
	case *object.Float: return obj.Value != 0.0
	case *object.String: return len(obj.Value) > 0
	case *object.List: return len(obj.Elements) > 0
	case *object.Tuple: return len(obj.Elements) > 0
	case *object.Dict: return len(obj.Pairs) > 0
	case *object.Set: return len(obj.Elements) > 0
	case *object.Bytes: return len(obj.Value) > 0
	default: return true // Other types are generally truthy
	}
}

// --- Collection Building ---
func (vm *VM) buildCollection(startIndex, endIndex int) ([]object.Object, error) {
	if startIndex < 0 || endIndex > vm.sp || startIndex > endIndex { return nil, fmt.Errorf("internal error: invalid indices for buildCollection") }
	elements := make([]object.Object, endIndex-startIndex)
	copy(elements, vm.stack[startIndex:endIndex])
	return elements, nil
}

// --- Indexing / Attributes / Iterators ---
func (vm *VM) executeIndexOperation(left, index object.Object) error {
	switch l := left.(type) {
	case *object.List: idxObj, ok := index.(*object.Integer); if !ok { return object.NewError("TypeError: list indices must be integers") }; return vm.executeListIndex(l, idxObj)
	case *object.Tuple: idxObj, ok := index.(*object.Integer); if !ok { return object.NewError("TypeError: tuple indices must be integers") }; return vm.executeTupleIndex(l, idxObj)
	case *object.Bytes: idxObj, ok := index.(*object.Integer); if !ok { return object.NewError("TypeError: bytes indices must be integers") }; return vm.executeBytesIndex(l, idxObj)
	case *object.String: idxObj, ok := index.(*object.Integer); if !ok { return object.NewError("TypeError: string indices must be integers") }; return vm.executeStringIndex(l, idxObj)
	case *object.Dict: return vm.executeDictIndex(l, index)
	default: return object.NewError("TypeError: '%s' object is not subscriptable", left.Type())
	}
}
func (vm *VM) executeListIndex(list *object.List, index *object.Integer) error { idx := index.Value; listLen := int64(len(list.Elements)); if idx < 0 { idx += listLen }; if idx < 0 || idx >= listLen { return object.NewError("IndexError: list index out of range") }; return vm.push(list.Elements[idx]) }
func (vm *VM) executeTupleIndex(tuple *object.Tuple, index *object.Integer) error { idx := index.Value; tupleLen := int64(len(tuple.Elements)); if idx < 0 { idx += tupleLen }; if idx < 0 || idx >= tupleLen { return object.NewError("IndexError: tuple index out of range") }; return vm.push(tuple.Elements[idx]) }
func (vm *VM) executeBytesIndex(bytesObj *object.Bytes, index *object.Integer) error { idx := index.Value; bytesLen := int64(len(bytesObj.Value)); if idx < 0 { idx += bytesLen }; if idx < 0 || idx >= bytesLen { return object.NewError("IndexError: bytes index out of range") }; return vm.push(&object.Integer{Value: int64(bytesObj.Value[idx])}) }
func (vm *VM) executeStringIndex(str *object.String, index *object.Integer) error { runes := []rune(str.Value); idx := index.Value; strLen := int64(len(runes)); if idx < 0 { idx += strLen }; if idx < 0 || idx >= strLen { return object.NewError("IndexError: string index out of range") }; return vm.push(&object.String{Value: string(runes[idx])}) }
func (vm *VM) executeDictIndex(dict *object.Dict, index object.Object) error { hashableKey, ok := index.(object.Hashable); if !ok { return object.NewError("TypeError: unhashable type: '%s'", index.Type()) }; dictMapKey, err := hashableKey.HashKey(); if err != nil { return object.NewError("TypeError: failed to hash key: %v", err) }; pair, ok := dict.Pairs[dictMapKey]; if !ok { return object.NewError("KeyError: %s", index.Inspect()) }; return vm.push(pair.Value) }

func (vm *VM) executeIndexAssignment(collection, index, value object.Object) error {
	switch coll := collection.(type) {
	case *object.List:
		idxObj, ok := index.(*object.Integer); if !ok { return object.NewError("TypeError: list indices must be integers") }
		idx := idxObj.Value; listLen := int64(len(coll.Elements)); if idx < 0 { idx += listLen }; if idx < 0 || idx >= listLen { return object.NewError("IndexError: list assignment index out of range") }
		coll.Elements[idx] = value; return nil
	case *object.Dict:
		key, ok := index.(object.Hashable); if !ok { return object.NewError("TypeError: unhashable type: '%s'", index.Type()) }
		hashKeyVal, err := key.HashKey(); if err != nil { return object.NewError("TypeError: failed to hash key: %v", err) }
		coll.Pairs[hashKeyVal] = object.DictPair{Key: index, Value: value}; return nil
	case *object.Tuple, *object.Bytes, *object.String: return object.NewError("TypeError: '%s' object does not support item assignment", collection.Type())
	default: return object.NewError("TypeError: '%s' object does not support item assignment", collection.Type())
	}
}

func (vm *VM) executeGetAttribute(obj object.Object, attrName string) error {
	switch instance := obj.(type) {
	case *Instance:
		if instance.Env != nil { if val, ok := instance.Env.Get(attrName); ok { return vm.push(val) } }
		vmClass := instance.Class; if vmClass == nil { return fmt.Errorf("internal error: vm.Instance has nil Class") }
		if methodClosure, methodOk := vmClass.Methods[attrName]; methodOk { return vm.push(&BoundMethod{Instance: instance, Method: methodClosure}) }
		return object.NewError("AttributeError: '%s' object has no attribute '%s'", vmClass.Name, attrName)
	case *object.Module:
		if val, ok := instance.Env.Get(attrName); ok { return vm.push(val) }
		return object.NewError("AttributeError: module '%s' has no attribute '%s'", instance.Name, attrName)
	default: return object.NewError("AttributeError: '%s' object has no attribute '%s'", obj.Type(), attrName)
	}
}

func (vm *VM) executeSetAttribute(obj object.Object, attrName string, value object.Object) error {
	switch instance := obj.(type) {
	case *Instance:
		if instance.Env == nil { instance.Env = object.NewEnvironment() }
		instance.Env.Set(attrName, value); return nil
	default: return object.NewError("AttributeError: '%s' object attribute assignment not supported", obj.Type())
	}
}

func (vm *VM) getObjectIterator(obj object.Object) (object.Iterator, error) {
	if inst, ok := obj.(*Instance); ok { if _, methodOk := inst.Class.Methods["__iter__"]; methodOk { return nil, object.NewError("NotImplementedError: vm OpGetIter doesn't support user-defined __iter__ yet") } }
	switch o := obj.(type) {
	case *object.List: return vm.newListIterator(o), nil
	case *object.Tuple: return vm.newTupleIterator(o), nil
	case *object.String: return vm.newStringIterator(o), nil
	case *object.Range: return vm.newRangeIterator(o), nil
	case *object.Dict: return vm.newDictIterator(o), nil
	case *object.Bytes: return vm.newBytesIterator(o), nil
	case *object.Set: return vm.newSetIterator(o), nil
	}
	if iterator, isIter := obj.(object.Iterator); isIter { return iterator, nil }
	return nil, object.NewError("TypeError: '%s' object is not iterable", obj.Type())
}
func (vm *VM) newListIterator(list *object.List) *object.GenericIterator   { idx := 0; return &object.GenericIterator{Source: "list", NextFn: func() (object.Object, bool) { if idx >= len(list.Elements) { return nil, true }; elem := list.Elements[idx]; idx++; return elem, false }} }
func (vm *VM) newTupleIterator(tuple *object.Tuple) *object.GenericIterator { idx := 0; return &object.GenericIterator{Source: "tuple", NextFn: func() (object.Object, bool) { if idx >= len(tuple.Elements) { return nil, true }; elem := tuple.Elements[idx]; idx++; return elem, false }} }
func (vm *VM) newStringIterator(str *object.String) *object.GenericIterator { runes := []rune(str.Value); idx := 0; return &object.GenericIterator{Source: "string", NextFn: func() (object.Object, bool) { if idx >= len(runes) { return nil, true }; r := runes[idx]; idx++; return &object.String{Value: string(r)}, false }} }
func (vm *VM) newRangeIterator(r *object.Range) *object.GenericIterator     { current, step, stop := r.Start, r.Step, r.Stop; finished := (step == 0) || (step > 0 && current >= stop) || (step < 0 && current <= stop); return &object.GenericIterator{Source: "range", NextFn: func() (object.Object, bool) { if finished { return nil, true }; val := current; current += step; if (step > 0 && current >= stop) || (step < 0 && current <= stop) { finished = true }; return &object.Integer{Value: val}, false }} }
func (vm *VM) newDictIterator(dict *object.Dict) *object.GenericIterator   { keys := make([]object.HashKey, 0, len(dict.Pairs)); idx := 0; for k := range dict.Pairs { keys = append(keys, k) }; /* TODO: Sort keys? */; return &object.GenericIterator{Source: "dict_keys", NextFn: func() (object.Object, bool) { if idx >= len(keys) { return nil, true }; keyHash := keys[idx]; idx++; pair, ok := dict.Pairs[keyHash]; if !ok { return object.NewError("InternalError: key disappeared during dict iteration"), true }; return pair.Key, false }} }
func (vm *VM) newBytesIterator(b *object.Bytes) *object.GenericIterator     { idx := 0; return &object.GenericIterator{Source: "bytes", NextFn: func() (object.Object, bool) { if idx >= len(b.Value) { return nil, true }; byteVal := b.Value[idx]; idx++; return &object.Integer{Value: int64(byteVal)}, false }} }
func (vm *VM) newSetIterator(s *object.Set) *object.GenericIterator       { elems := make([]object.Object, 0, len(s.Elements)); idx := 0; for _, elem := range s.Elements { elems = append(elems, elem) }; /* TODO: Sort elements? */; return &object.GenericIterator{Source: "set", NextFn: func() (object.Object, bool) { if idx >= len(elems) { return nil, true }; elem := elems[idx]; idx++; return elem, false }} }


// --- Call Helpers ---
// --- callBuiltin with Logging ---
func (vm *VM) callBuiltin(builtin *object.Builtin, numArgs int) error {
    // Args start at sp - numArgs. Builtin is at sp - numArgs - 1.
	argsSliceStart := vm.sp - numArgs; if argsSliceStart < 0 { return fmt.Errorf("stack underflow calculating args start for builtin call (sp=%d, numArgs=%d)", vm.sp, numArgs) }
    builtinSlot := argsSliceStart - 1; if builtinSlot < 0 { return fmt.Errorf("stack underflow locating builtin for call (sp=%d, numArgs=%d)", vm.sp, numArgs)}

	args := make([]object.Object, numArgs); copy(args, vm.stack[argsSliceStart:vm.sp])
	builtinName := "<unknown_builtin>" // Try to get name later if needed

    fmt.Printf(">>> DEBUG VM: callBuiltin START (%s) with %d args. SP before call: %d\n", builtinName, numArgs, vm.sp) // DEBUG

	// Pop args AND builtin, leave space for result
    vm.sp = builtinSlot
    for i := vm.sp; i < argsSliceStart+numArgs; i++ { // Clear popped slots
        vm.stack[i] = nil
    }

	syncCtx := &SynchronousVMContext{vm: vm} // Use synchronous context for direct calls
	result := builtin.Fn(syncCtx, args...) // Execute builtin

    fmt.Printf("<<< DEBUG VM: callBuiltin END (%s). Result: %s. SP after call (before push): %d\n", builtinName, objInspect(result), vm.sp) // DEBUG

	if pyErr, ok := result.(object.Object); ok && object.IsError(pyErr) {
        return pyErr.(error) // Return Pylearn error as Go error
    }

    // Push the result onto the stack (where the builtin was)
	errPush := vm.push(result); if errPush != nil { return fmt.Errorf("callBuiltin push result error: %w", errPush) }
    fmt.Printf(">>> DEBUG VM: callBuiltin (%s) PUSHED result. SP after push: %d\n", builtinName, vm.sp) // DEBUG
	return nil // Success
}


func (vm *VM) callClass(class *Class, numArgs int) error {
	// Args are already on stack: [..., arg1, argN] SP points after argN
	// Class object is below args: [..., ClassObj, arg1, argN]
	argsSliceStart := vm.sp - numArgs
	classSlot := argsSliceStart - 1
	if classSlot < 0 { return fmt.Errorf("stack underflow locating class object for call") }

	// Create the instance
	instance := &Instance{Class: class, Env: object.NewEnvironment()}

	// Replace the Class object on the stack with the new Instance
	vm.stack[classSlot] = instance
	// Pop the arguments (we'll push them back for __init__)
	args := make([]object.Object, numArgs)
	copy(args, vm.stack[argsSliceStart:vm.sp]) // Copy args before changing SP
	vm.sp = argsSliceStart                     // Pop args, SP now points after instance
	for i := vm.sp; i < argsSliceStart+numArgs; i++ { vm.stack[i] = nil } // Clear slots

	fmt.Printf(">>> DEBUG VM: callClass START (%s): Created instance %s. SP=%d\n", class.Name, objInspect(instance), vm.sp) // SP points *after* instance

	// Find and potentially call __init__
	initMethodClosure, initExists := class.Methods["__init__"]
	if initExists {
		fmt.Printf(">>> DEBUG VM: callClass (%s): Found __init__. Preparing call.\n", class.Name)

		// 1. Push the __init__ closure itself
		err := vm.push(initMethodClosure)
		if err != nil { vm.sp = classSlot; return fmt.Errorf("stack overflow pushing __init__ closure: %w", err) } // Restore SP to before instance

		// 2. Push the instance (self) as the first argument
		err = vm.push(instance) // Push self
		if err != nil { vm.sp = classSlot + 1; vm.pop(); return fmt.Errorf("stack overflow pushing self for __init__: %w", err)} // Pop closure

        // 3. Push the original arguments
        // spBeforeArgs := vm.sp
		for i, arg := range args {
			err = vm.push(arg)
			if err != nil {
                // Clean up pushed args, self, closure
                vm.sp = classSlot + 1 // Restore SP to after instance slot
                return fmt.Errorf("stack overflow pushing arg %d for __init__: %w", i, err)
            }
		}

        // 4. Call the closure. numArgs for __init__ is original numArgs + 1 (for self)
		totalInitArgs := numArgs + 1
		err = vm.callClosure(initMethodClosure, totalInitArgs)
		if err != nil {
            vm.sp = classSlot + 1 // Restore SP to after instance slot on call setup error
            return fmt.Errorf("error setting up __init__ call: %w", err)
        }
		fmt.Printf(">>> DEBUG VM: callClass (%s): __init__ frame pushed.\n", class.Name)
        // Return nil, the Run loop continues in the __init__ frame
        return nil // Signal success, ipWasSet=false (frame changed)

	} else {
		fmt.Printf(">>> DEBUG VM: callClass (%s): No __init__ found.\n", class.Name)
		if numArgs > 0 {
            // If no __init__, pop the instance we put on stack and return error
            vm.sp = classSlot
            vm.stack[vm.sp] = nil
            return object.NewError("TypeError: %s() takes no arguments when __init__ is absent (%d given)", class.Name, numArgs)
        }
		// If no __init__ and no args, the instance is already correctly on the stack at classSlot.
        // SP is already pointing after it (vm.sp = argsSliceStart = classSlot + 1).
		fmt.Printf(">>> DEBUG VM: callClass (%s): Instance %s left on stack. SP=%d\n", class.Name, objInspect(instance), vm.sp)
        return nil // Success, ipWasSet=false (no frame change, normal advance)
	}
}


// --- callClosure ---
func (vm *VM) callClosure(cl *Closure, numArgs int) error {
	if cl == nil || cl.Fn == nil { return fmt.Errorf("internal vm error: callClosure called with nil closure/function") }
	expectedParams := cl.Fn.NumParameters; funcName := cl.Fn.Name; if funcName == "" { funcName = "<lambda>" }
	if numArgs != expectedParams { return object.NewError("TypeError: %s() takes %d positional arguments but %d were given", funcName, expectedParams, numArgs) }

    // Calculate Base Pointer: Points to the slot holding the *closure* being called.
    // Stack: [..., closure, arg1, ..., argN] <-- SP points here
    // BP = SP - numArgs - 1
	basePointer := vm.sp - numArgs - 1
	if basePointer < 0 { return fmt.Errorf("stack underflow calculating base pointer for call (sp=%d, numArgs=%d)", vm.sp, numArgs) }

    fmt.Printf(">>> DEBUG VM: callClosure START (%s): Args=%d, SP=%d, BP=%d\n", funcName, numArgs, vm.sp, basePointer) // DEBUG
	newFrame := NewFrame(cl, basePointer) // IP starts at -1
	err := vm.pushFrame(newFrame); if err != nil { return err } // Pushes frame, increments frameIndex

	// Allocate space for locals NOT including parameters
    numLocalsTotal := cl.Fn.NumLocals; numParameters := cl.Fn.NumParameters
	numNewLocalsToAllocate := numLocalsTotal - numParameters
	if numNewLocalsToAllocate < 0 { vm.popFrame(); return fmt.Errorf("internal compiler error: NumLocals %d < NumParameters %d for %s", numLocalsTotal, numParameters, funcName) }
    fmt.Printf(">>> DEBUG VM: callClosure (%s): Allocating %d new locals (total %d, params %d)\n", funcName, numNewLocalsToAllocate, numLocalsTotal, numParameters) // DEBUG
	for i := 0; i < numNewLocalsToAllocate; i++ {
		err = vm.push(object.NULL) // push increments vm.sp
		if err != nil {
            // If overflow allocating locals, pop frame and restore SP
            vm.popFrame()
            vm.sp = basePointer + numArgs + 1 // Restore SP to before local alloc
            return fmt.Errorf("stack overflow allocating locals: %w", err)
        }
	}
    fmt.Printf(">>> DEBUG VM: callClosure END (%s): Frame pushed. New SP=%d\n", funcName, vm.sp) // DEBUG
	return nil // Success
}

func (vm *VM) callBoundMethod(boundMethod *BoundMethod, numArgs int) error {
	methodClosure := boundMethod.Method; instance := boundMethod.Instance
    methodName := "<method>"
    if methodClosure != nil && methodClosure.Fn != nil { methodName = methodClosure.Fn.Name }
	argsSliceStart := vm.sp - numArgs; if argsSliceStart < 0 || argsSliceStart-1 < 0 { return fmt.Errorf("stack underflow preparing for bound method call") }
	args := make([]object.Object, numArgs); copy(args, vm.stack[argsSliceStart:vm.sp])
	vm.sp = argsSliceStart - 1 // Pop args AND BoundMethod
    fmt.Printf(">>> DEBUG VM: callBoundMethod START (%s): Self=%s, Args=%d, SP=%d\n", methodName, objInspect(instance), numArgs, vm.sp) // DEBUG

	// Must push Closure *then* self to match callClosure expectation
	err := vm.push(methodClosure); if err != nil { vm.sp = argsSliceStart - 1; return fmt.Errorf("stack overflow pushing method closure: %w", err) }
	err = vm.push(instance); if err != nil { vm.sp = argsSliceStart - 1; vm.pop(); return fmt.Errorf("stack overflow pushing self for method call: %w", err) }
	spBeforeArgs := vm.sp
	for _, arg := range args { err = vm.push(arg); if err != nil { vm.sp = spBeforeArgs; vm.pop(); vm.pop(); vm.sp = argsSliceStart - 1; return fmt.Errorf("stack overflow pushing args for method call: %w", err) } }
	err = vm.callClosure(methodClosure, numArgs+1); if err != nil { vm.sp = argsSliceStart - 1; return fmt.Errorf("error during bound method call setup: %w", err) }
    fmt.Printf(">>> DEBUG VM: callBoundMethod END (%s): Frame pushed.\n", methodName) // DEBUG
	return nil
}
